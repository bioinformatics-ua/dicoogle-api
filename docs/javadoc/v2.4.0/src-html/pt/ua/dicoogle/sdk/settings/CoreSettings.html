019package pt.ua.dicoogle.sdk.settings;
020
021/**
022 * Contains the list of Dicoogle Settings 
023 * 
024 * <p>
025 * <b> This list can be accessed by the Plugins</b>
026 * </p>
027 * @author Luís A. Bastião Silva <bastiao@ua.pt>
028 */
029public interface CoreSettings
030{
031    /**
032     * This option is related with the de-identification of the patient data
033     * If this option is enable the documents will be indexed and shown in an
034     * anonymous format. 
035     * @return boolean true if anonymize is enable, false otherwise. 
036     */
037    public boolean isIndexAnonymous();
038    
039    /**
040     * If this option is enable, the database will store a thumbnail of each 
041     * image 
042     * @return boolean true if is necessary to create thumbnail, false otherwise
043     */
044    public boolean isThumbnails();
045    /**
046     * Get the matrix size of the thumbnail
047     * 
048     * @return String matrix size in the format numberXnumber, for example,
049     * 8x8 or 16x16 pixels 
050     */
051    public String getThumbnailsMatrix();
052    
053}
