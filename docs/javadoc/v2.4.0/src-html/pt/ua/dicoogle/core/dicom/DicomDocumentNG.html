078     * Convert the byte array to an int starting from the given offset.
079     *
080     * @param b The byte array
081     * @param offset The array offset
082     * @return The integer
083     */
084    public static long byteArrayToInt(byte[] b, int offset) {
085        long value = 0;
086        for (int i = 0; i < b.length; i++) {
087            int shift = (4 - 1 - i) * 8;
088            value += (b[i + offset] & 0x000000FF) << shift;
089        }
090        return value;
091    }

093    /**
094     * Converts a 4 byte array of unsigned bytes to an long
095     *
096     * @param b an array of 4 unsigned bytes
097     * @return a long representing the unsigned int
098     */
099    public static final long unsignedIntToLong(byte[] b) {
100        long l = 0;
101        l |= b[0] & 0xFF;
102        l <<= 8;
103        l |= b[1] & 0xFF;
104        l <<= 8;
105        l |= b[2] & 0xFF;
106        l <<= 8;
107        l |= b[3] & 0xFF;
108        return l;
109    }

123    public static final int unsignedShortToInt(byte[] b) {
124        int i = 0;
125        i |= b[0] & 0xFF;
126        i <<= 8;
127        i |= b[1] & 0xFF;
128        return i;

153    public static String byteArrayToHexString(byte[] b) {
154        StringBuffer sb = new StringBuffer(b.length * 2);
155        for (int i = 0; i < b.length; i++) {
156            int v = b[i] & 0xff;
157            if (v < 16) {
158                sb.append('0');
159            }
160            sb.append(Integer.toHexString(v));
161        }
162        return sb.toString().toUpperCase();
163    }

165    public static String getValue(DicomElement element) {
166        String value;
167        
168        if (!DicomDocumentNG.isBinaryFields(element.vr())) {
169            value = null;
170            Charset utf8charset = Charset.forName("UTF-8");
171            Charset iso88591charset = Charset.forName("iso8859-1");
172            byte[] values = element.getBytes();
173            ByteBuffer inputBuffer = ByteBuffer.wrap(values);
174
175            // decode UTF-8
176            CharBuffer data = iso88591charset.decode(inputBuffer);
177
178            // encode ISO-8559-1
179            ByteBuffer outputBuffer = utf8charset.encode(data);
180            
181            byte[] outputData = outputBuffer.array();
182            try {
183                value = new String(outputData, "UTF-8");
184            } catch (UnsupportedEncodingException ex) {
185                LoggerFactory.getLogger(DicomDocumentNG.class).error(ex.getMessage(), ex);
186            }
187            /*
188             * 
189             *  ASCII         "ISO_IR 6"    =>  "UTF-8"
190             UTF-8         "ISO_IR 192"  =>  "UTF-8"
191             ISO Latin 1   "ISO_IR 100"  =>  "ISO-8859-1"
192             ISO Latin 2   "ISO_IR 101"  =>  "ISO-8859-2"
193             ISO Latin 3   "ISO_IR 109"  =>  "ISO-8859-3"
194             ISO Latin 4   "ISO_IR 110"  =>  "ISO-8859-4"
195             ISO Latin 5   "ISO_IR 148"  =>  "ISO-8859-9"
196             Cyrillic      "ISO_IR 144"  =>  "ISO-8859-5"
197             Arabic        "ISO_IR 127"  =>  "ISO-8859-6"
198             Greek         "ISO_IR 126"  =>  "ISO-8859-7"
199             Hebrew        "ISO_IR 138"  =>  "ISO-8859-8"
200             */
201        } else {
202            if (element.vr() == VR.FD && element.getBytes().length == 8) {
203                double tmpValue = element.getDouble(true);
204                value = String.valueOf(tmpValue);
205            } else if (element.vr() == VR.FL && element.getBytes().length == 4) {
206                float tmpValue = element.getFloat(true);
207                value = String.valueOf(tmpValue);                
208            } else if (element.vr() == VR.UL && element.getBytes().length == 4) {
209                //long tmpValue = unsignedIntToLong(element.getBytes());
210                long tmpValue = element.getInt(true);
211                value = String.valueOf(tmpValue);
212            } else if (element.vr() == VR.US && element.getBytes().length == 2) {
213                short[] tmpValue = element.getShorts(true);
214                value = String.valueOf(tmpValue[0]);
215            } else if (element.vr() != VR.US) {
216                long tmpValue = byteArrayToInt(element.getBytes());
217                value = String.valueOf(tmpValue);
218            } else {
219                int tmpValue = element.getInt(true);
220                value = String.valueOf(tmpValue);
221            }
222        }
223        return value;
224    }

226    public static long toLong(byte[] data) {
227        if (data == null || data.length != 8) {
228            return 0x0;
229        }
230        // ----------
231        return (long) ( // (Below) convert to longs before shift because digits
232                //         are lost with ints beyond the 32-bit limit
233                (long) (0xff & data[0]) << 56
234                | (long) (0xff & data[1]) << 48
235                | (long) (0xff & data[2]) << 40
236                | (long) (0xff & data[3]) << 32
237                | (long) (0xff & data[4]) << 24
238                | (long) (0xff & data[5]) << 16
239                | (long) (0xff & data[6]) << 8
240                | (long) (0xff & data[7]) << 0);
241    }

244        if (data == null || data.length != 8) {
245            return 0x0;

247    // ---------- simple:
248        
249        return Double.longBitsToDouble(toLong(data));

252    public static double[] toDoubleA(byte[] data) {
253        if (data == null) {
254            return null;
255        }
256        // ----------
257        if (data.length % 8 != 0) {
258            return null;
259        }
260        double[] dbls = new double[data.length / 8];
261        for (int i = 0; i < dbls.length; i++) {
262            dbls[i] = toDouble(new byte[]{
263                data[(i * 8)],
264                data[(i * 8) + 1],
265                data[(i * 8) + 2],
266                data[(i * 8) + 3],
267                data[(i * 8) + 4],
268                data[(i * 8) + 5],
269                data[(i * 8) + 6],
270                data[(i * 8) + 7],});
271        }
272        return dbls;
273    }

275    public static final int byteArrayToInt2(byte[] b) {
276        return (b[0] << 24)
277                + ((b[1] & 0xFF) << 16)
278                + ((b[2] & 0xFF) << 8)
279                + (b[3] & 0xFF);
280    }

282    public String convertStringToHex(String str) {
283        
284        char[] chars = str.toCharArray();
285        
286        StringBuffer hex = new StringBuffer();
287        for (int i = 0; i < chars.length; i++) {
288            hex.append(Integer.toHexString((int) chars[i]));
289        }
290        
291        return hex.toString();
292    }

294    public String anonymousFilter(String tag, String value) {
295        String result = value;
296        if (value == null) {
297            return value;
298        }
299        if (tag.equals("PatientName")) {
300            result = MatchTable.getInstance().getName(value);
301        } else if (tag.equals("PatientID")) {
302            result = MatchTable.getInstance().getPatientID(value);
303        } else if (tag.equals("AccessionNumber")) {
304            result = MatchTable.getInstance().getAccessionNumber(value);
305        }
306        return result;
307    }

309    /**
310     * Retrieve information of DICOM file based on XML and loaded tags
311     *
312     * @param f
313     * @return
314     * @throws
315     * pt.ua.dicoogle.core.index.lucene.LuceneSupport.DocumentHandlerException
316     */
317    public IDoc getDocument(File f) throws DocumentHandlerException, FileAlreadyExistsException {
318        
319        
320        Iterable<SearchResult> it = search.search("uri:\"" + f.getAbsolutePath() + "\"");
321        
322        
323        if (it.iterator().hasNext() ) {
324            System.out.println("File already exists" );
325            throw new FileAlreadyExistsException();
326        }
327        
328        System.out.println("File  does not exists.." );
329        IDoc _doc = null;
330        
331        tagStruct = TagsStruct.getInstance();
332        
333        FileInputStream fis = null;
334        DicomInputStream dis = null;
335        
336        BufferedInputStream bis = null;
337        
338        try {
339            if (f.getAbsolutePath().endsWith(".dll") || f.getAbsolutePath().endsWith(".exe") || f.getName().startsWith(".")) {
340                return null;
341            }
342            
343            fis = new FileInputStream(f);
344            if (f.getAbsolutePath().endsWith(".gz")) {
345                bis = new BufferedInputStream(new GZIPInputStream(fis));
346            } else {
347                bis = new BufferedInputStream(fis);
348            }
349            
350            dis = new DicomInputStream(bis);
351            //dis = new DicomInputStream(bis);
352
353            //TODO: fix me
354            //dis.setFileSize(f.length());
355            dis.setHandler(new StopTagInputHandler(Tag.PixelData));
356
357            //DebugManager.getInstance().debug(f.getAbsolutePath());
358            DicomObject d = dis.readDicomObject();
359            _doc = new DicomDocument();
360
361            /**
362             * Verify if SOPInstanceUID exists If it exists it will verify
363             * columns + rows. If it is a big image it will be re-index
364             * otherwise it will be skipped.
365             */
366            String SOPInstanceUID = d.getString(Tag.SOPInstanceUID);
367            
368            DicomElement e1 = d.get(Tag.Columns);
369            DicomElement e2 = d.get(Tag.Rows);
370            
371            if (e1 != null && e2 != null) {
372                String columns = getValue(e1);
373                String rows = getValue(e2);
374                if (rows != null && !rows.equals("") && columns != null && !columns.equals("")) {
375                    String query = "SOPInstanceUID:" + SOPInstanceUID + " AND (Columns:Float:[" + columns + " TO 65536] AND Rows:Float:[" + rows + " TO 65536])";
376                    it = search.search(query);
377                    if (it.iterator().hasNext()) {
378
379                        //System.out.println("SOPInstanceUID already exists: " + SOPInstanceUID);
380                        throw new FileAlreadyExistsException();
381                        //return null;
382                    }
383                }
384            }
385            
386            if (sops.contains(SOPInstanceUID)) {
387                System.out.println("SOPInstanceUID already exists: " + SOPInstanceUID);
388                throw new FileAlreadyExistsException();
389            } else {
390                sops.add(SOPInstanceUID);
391            }
392
393            /**
394             * How it it works? Actually Lucene 2.4.X doesn't support added
395             * hidden fields, so the stretegie it is following: - DIM fields
396             * specified on XML are already indexed - Other fields was getted
397             * dinamic way (It depends of Settings) - Thumbnails are indexed, if
398             * enabled on Settings
399             */
400            /**
401             * *
402             *
403             * DIM Fields is mandatory -- No need to check on file XML file if
404             * it is really enabled
405             *
406             *
407             */
408            String data = null;
409            
410            for (TagValue tag : tagStruct.getDIMFields()) {
411                int key = tag.getTagNumber();
412                data = null;
413                DicomElement e = d.get(key);
414                if (settings.isIndexAnonymous()) {
415                    data = this.anonymousFilter(tag.getAlias(), d.getString(key));
416                } else {
417                    if (e != null) {
418                        data = getValue(e);
419                    }
420                }
421
422                /**
423                 * If it is null should be passed has a empty string, it's bad
424                 * idea pass null to Lucene.
425                 */
426                if (data == null) {
427                    data = "";
428                }
429                addField(_doc, d.vrOf(key), tag.getAlias(), data);
430            }
431
432            //DebugManager.getInstance().debug(">>>> DIM fields was indexed");
433            String otherToIndex = "";
434
435            /**
436             * Other fields is not mandatory It is needed to check if
437             *
438             */
439            for (TagValue tag : tagStruct.getOtherFields()) {
440                data = d.getString(tag.getTagNumber());
441                /**
442                 * If it is null should be passed has a empty string, it's bad
443                 * idea pass null to Lucene.
444                 */
445                if (data == null) {
446                    data = "";
447                }
448                
449                VR v = VR.valueOf(Integer.parseInt(convertStringToHex(tag.getVR()), 16));
450                
451                addField(_doc, v, tag.getAlias(), data);
452                otherToIndex = otherToIndex + " " + data;
453            }
454
455            //DebugManager.getInstance().debug(">>>> DIM fields was indexed");
456            /**
457             * Other or Modalities need a deep search on DICOM Document And it
458             * produces a delay indexing
459             */
460            if (tagStruct.isDeepSearchModalitiesEnabled()) {
461
462                /**
463                 * Now verify if the modality of this document is enable in
464                 * setting file
465                 */
466                if (tagStruct.isModalityEnable(d.getString(Tag.Modality)) || tagStruct.isIndexAllModalitiesEnabled()) {
467                    List<String> _list = getRecursiveDicomElement(_doc, d, "", -1);
468                    otherToIndex = _list.get(0);
469                }
470            }
471            //DebugManager.getInstance().debug("Indexing others .. Modalities");
472            _doc.add("others", otherToIndex);
473            
474            if (settings != null && indexStoreThumbnails()) {
475                int matrix = Integer.parseInt(settings.getThumbnailsMatrix());
476                ByteArrayOutputStream jpgMem = Dicom2JPEG.Dicom2MemJPEG(f, matrix);
477                //DebugManager.getInstance().log("SIZE: " + jpgMem.toByteArray().length);
478                if (jpgMem != null) {
479                    _doc.add("Thumbnail", jpgMem.toByteArray());
480                }
481            }
482        } catch (FileAlreadyExistsException ex) {
483            throw new FileAlreadyExistsException();
484        } catch (Exception ex) {
485            ex.printStackTrace();
486            //DebugManager.getInstance().log("Index error: " + f.getAbsolutePath() + " : " + 
487            //         "\n");
488            //if (DebugManager.getInstance().isDebug())
489            //     ex.printStackTrace();
490            // DebugManager.getInstance().log(f.getAbsolutePath() + " : " + 
491            //         ex.toString() + "\n");
492            
493            _doc = null;
494            
495        } finally {
496            
497            try {
498                dis.close();
499                fis.close();
500                bis.close();
501            } catch (Exception ex) {
502                ex.printStackTrace();
503                _doc = null;
504                
505            }
506        }
507
508        //long end = System.currentTimeMillis();
509        //System.out.println("Execution time was "+(end-start)+" ms.");
510        System.out.println("_DOC: " + _doc);
511        return _doc;
512        
513    }

515    private List<String> getRecursiveDicomElement(IDoc _doc, DicomObject d, String prefix,int nItems) {
516        
517        String otherToIndex = "";
518        String tagList = "";
519        List<String> tmp = new ArrayList<String>();
520        // Hard heuristic just to be sure that application will be not running forever
521        if (prefix.length() > 512) {
522            
523            tmp.add(otherToIndex);
524            tmp.add(tagList);
525            
526        }
527        Map<String, String> sequences = new HashMap<String, String>();
528        
529        DictionaryAccess dictionaryDicom = DictionaryAccess.getInstance();
530        Iterator<DicomElement> it = d.iterator();
531                
532        int i = 0;
533        while (it.hasNext()) {
534            DicomElement dcm = it.next();
535            int tmpTag = dcm.tag();
536            //System.out.println(tmpTag);
537            
538            if (!tagStruct.containsTag(tmpTag)) {
539                String tagName = d.nameOf(dcm.tag());
540                
541                String newTag = dictionaryDicom.tagName(dcm.tag());
542                
543                if (dcm.hasItems()) {
544                    
545                    newTag = dictionaryDicom.tagName(dcm.tag());
546                    
547                    String prefixAux = prefix + newTag + "_";
548                    
549                    if (dcm.countItems() > 0) {
550                        List<String> list = getRecursiveDicomElement(_doc, dcm.getDicomObject(0), prefixAux, dcm.countItems());
551                        
552                        otherToIndex = otherToIndex + " " + list.get(0);
553                        tagList = tagList + " " + list.get(1);
554                        
555                        sequences.put(newTag, list.get(1));                        
556                    }
557                    
558                } /**
559                 *
560                 * Drop the non-search-valid fields (Pixel data etc)
561                 *
562                 */
563                else if (dcm.vr() != VR.OB && dcm.vr() != VR.OW && !tagName.equals("?") && dictionaryDicom.tagName(dcm.tag()) != null) {
564                    String value;
565                    
566                    value = getValue(dcm);
567                    
568                    if (value != null) {
569                        newTag = dictionaryDicom.tagName(dcm.tag());
570                        tagList = tagList + " " + prefix + newTag;
571                        if (newTag != null) {
572                            addField(_doc, dcm.vr(), prefix + newTag, value);
573                        }
574                        otherToIndex = otherToIndex + " " + value;
575                    }
576                }
577                
578            }
579            i++;
580            //if (i!=0 && i>=nItems)
581            //    break;
582        }
583        //System.out.println("FIM DA RECURSIVIDADE");
584        // There is a sequence
585        
586        for (String s : sequences.keySet()) {
587            //System.out.println("SQ: " + s +", val: " + sequences.get(s));
588            addField(_doc, VR.ST, s, sequences.get(s));            
589        }
590        
591        tmp = new ArrayList<String>();
592        
593        tmp.add(otherToIndex);
594        tmp.add(tagList);
595        
596        return tmp;
597    }

611    private void addField(IDoc _doc, VR vr, String tag, String value) {
612        if (_doc == null) {
613            return;
614        }
615        if (tag == null) {
616            return;
617        }
618        if (value == null) {
619            value = "";
620        }
621        if (vr == VR.IS || vr == VR.DS || vr == VR.US || vr == VR.FD || vr == VR.FD) {
622            
623            try {
624                
625                Float _v = Float.valueOf(value);
626                _doc.add(tag, _v);
627                //System.out.println("Tag: " + tag + " : " + value);
628            } catch (NumberFormatException ex) {
629                _doc.add(tag, value);
630            }
631        } else {
632            _doc.add(tag, value);
633        }
634    }
