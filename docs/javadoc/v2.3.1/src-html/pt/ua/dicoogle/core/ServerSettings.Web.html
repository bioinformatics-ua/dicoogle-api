869    public boolean removeMoveDestination(String AETitle, String ipAddr, int port)
870    {
871        for(int i=0;i<dest.size(); i++)
872        {
873                MoveDestination mv = dest.get(i);
874                if(mv.getAETitle().equals(AETitle) && mv.getIpAddrs().equals(ipAddr) && mv.getPort() == port)
875                {
876                        dest.remove(i);
877                        return true;
878                }
879                        
880        }
881        return false;
882    }

1210    @Override
1211    public ArrayList<String> getNetworkInterfacesNames()
1212    {
1213        ArrayList<String> interfaces = new ArrayList<String>();
1214        Enumeration<NetworkInterface> nets = null;
1215        try
1216        {
1217            nets = NetworkInterface.getNetworkInterfaces();
1218        } catch (SocketException ex)
1219        {
1220            ex.printStackTrace();
1221        }
1222
1223        
1224        for (NetworkInterface netint : Collections.list(nets))
1225        {
1226            try
1227            {
1228                if (!netint.isLoopback())
1229                {
1230                    Enumeration<InetAddress> addresses = netint.getInetAddresses();
1231                    while (addresses.hasMoreElements())
1232                    {
1233                        if (Inet4Address.class.isInstance(addresses.nextElement()))
1234                        {
1235                            interfaces.add(netint.getDisplayName());
1236                        }
1237                    }
1238                }
1239            } catch (SocketException ex)
1240            {
1241                LoggerFactory.getLogger(ServerSettings.class).error(ex.getMessage(), ex);
1242            }
1243        }
1244        return interfaces;
1245    }

1252        {
1253            nets = NetworkInterface.getNetworkInterfaces();
1254        } catch (SocketException ex)
1255        {
1256            ex.printStackTrace();
1257        }
1258
1259        for(NetworkInterface netint : Collections.list(nets))
1260        {
1261            if(netint.getDisplayName().compareTo(this.networkInterfaceName) == 0)
1262            {
1263                Enumeration<InetAddress> addresses = netint.getInetAddresses();
1264                while(addresses.hasMoreElements())
1265                {
1266                    InetAddress address = addresses.nextElement();
1267                    if(Inet4Address.class.isInstance(address))
1268                    {
1269                        return address.getHostAddress();
1270                    }
1271                }
1272                return null;
1273            }
1274        }
1275        return null;
