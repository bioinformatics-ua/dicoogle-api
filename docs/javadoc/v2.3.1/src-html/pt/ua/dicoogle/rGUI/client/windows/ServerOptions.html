350    private void loadTagsDIMFields(DefaultMutableTreeNode top) {
351        try {
352            //DebugManager.getInstance().debug("Getting DIM Fields");
353
354            DefaultMutableTreeNode group = null;
355            DefaultMutableTreeNode subGroup = null;
356
357            HashMap<String, ArrayList<TagValue>> groupTable = indexOptions.getDIMFields();
358
359            /** Create nodes */
360            for (String g : groupTable.keySet()) {
361                group = new DefaultMutableTreeNode(g);
362                top.add(group);
363                //DebugManager.getInstance().debug("Adding new group: " + g);
364                for (TagValue t : groupTable.get(g)) {
365                    //DebugManager.getInstance().debug("Adding sub-group: " + t.getAlias());
366                    subGroup = new DefaultMutableTreeNode("(" + TagValue.getSubgroup(t.getTagNumber()) + ")  " + t.getAlias());
367                    group.add(subGroup);
368                }
369            }
370        } catch (RemoteException ex) {
371            LoggerFactory.getLogger(ServerOptions.class).error(ex.getMessage(), ex);
372        }
373    }

394    private void loadTagsOthers(DefaultMutableTreeNode top) {
395        try {
396            //DebugManager.getInstance().debug(">> Filling the Others Tags in MainWindow Notebook");
397
398            DefaultMutableTreeNode group = null;
399            DefaultMutableTreeNode subGroup = null;
400
401            HashMap<String, ArrayList<TagValue>> groupTable = indexOptions.getManualFields();
402
403            /** Finally Create nodes */
404            for (String g : groupTable.keySet()) {
405                //DebugManager.getInstance().debug(">> Adding the Others Tags in MainWindow Notebook");
406                group = new DefaultMutableTreeNode(g);
407                top.add(group);
408                for (TagValue t : groupTable.get(g)) {
409                    subGroup = new DefaultMutableTreeNode("(" + TagValue.getSubgroup(t.getTagNumber()) + ")  " + t.getAlias());
410                    group.add(subGroup);
411                }
412            }
413        } catch (RemoteException ex) {
414            LoggerFactory.getLogger(ServerOptions.class).error(ex.getMessage(), ex);
