424    /**
425     * @return ArrayList<String> with selected items in the tree
426     */
427    private ArrayList<String> getSelectedLocalFiles() {
428        ArrayList<String> files = new ArrayList<String>();
429
430        TreePath path = jTreeResults.getSelectionPath();
431
432        // Tree Root is not permited
433        if (path == null || path.getPathCount() < 2) {
434            return null;
435        }
436
437        jTreeResults.expandPath(path);
438
439        if (path != null) {
440            DefaultMutableTreeNode node = (DefaultMutableTreeNode) path.getLastPathComponent();
441
442            // recieves all childs
443            ArrayList<DefaultMutableTreeNode> childs = getLocalLeafs(node, path);
444
445            Iterator<DefaultMutableTreeNode> it = childs.iterator();
446
447            // converts TreeNodes to filePaths
448            while (it.hasNext()) {
449                Object obj = it.next().getUserObject();
450
451                if (SearchResult.class.isInstance(obj)) {
452                    files.add(obj.toString());
453                }
454            }
455        }
456
457        return files;
458    }

460    private ArrayList<DefaultMutableTreeNode> getLocalLeafs(DefaultMutableTreeNode node, TreePath path) {
461        ArrayList<DefaultMutableTreeNode> list = new ArrayList<DefaultMutableTreeNode>();
462        TreePath temp;
463
464        if (node.isLeaf()) {
465            list.add(node);
466        } else {
467            Enumeration<DefaultMutableTreeNode> en = node.children();
468
469            while (en.hasMoreElements()) {
470                DefaultMutableTreeNode elem = en.nextElement();
471
472                temp = path.pathByAddingChild(elem);
473                jTreeResults.expandPath(temp);
474
475                list.addAll(getLocalLeafs(elem, temp));
476            }
477        }
478
479        return list;
480    }
