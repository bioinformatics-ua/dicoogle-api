209        /**
210         * Tries to set the supplied settings as the new settings.
211         *
212         * @param settings a Map containing the new settings.
213         * @return true if all settings were successfully changed to the ones supplied, false otherwise (no settings changed).
214         */
215        public boolean setIndexingSettings(HashMap<String, Object> settings)
216        {
217                // validate the settings
218                boolean valid = false;
219                try
220                {
221                        valid = tryIndexingSettings(settings);
222                }
223                catch (InvalidSettingValueException ex)
224                {
225                        // at least one of the setting is invalid, abort
226                        return false;
227                }
228                if (! valid) //settings not valid due to unknown reason, also abort
229                        return false;
230
231                // all the settings were correctly validated, so apply them all
232                cfg.setDicoogleDir(((ServerDirectoryPath) settings.get(SETTING_DIRECTORY_MONITORIZATION_NAME)).getPath());
233                cfg.setMonitorWatcher(((Boolean) settings.get(SETTING_DIRECTORY_WATCHER_NAME)).booleanValue());
234                cfg.setIndexerEffort(((RangeInteger) settings.get(SETTING_INDEXING_EFFORT_NAME)).getValue());
235                cfg.setThumbnailsMatrix((String) settings.get(SETTING_THUMBNAILS_SIZE_NAME));
236                cfg.setSaveThumbnails(((Boolean) settings.get(SETTING_SAVE_THUMBNAILS_NAME)).booleanValue());
237                cfg.setIndexZIPFiles(((Boolean) settings.get(SETTING_INDEX_ZIP_FILES_NAME)).booleanValue());
238
239                // return success
240                return true;
241        }

248         * Returns a Map containing the current settings.

252        public HashMap<String, Object> getAccessListSettings()
253        {
254                HashMap<String, Object> settings = new LinkedHashMap<String, Object>();
255
256                settings.put(SETTING_SERVER_AE_TITLE_NAME, cfg.getAE());
257                settings.put(SETTING_PERMIT_ALL_AE_TITLES_NAME, new Boolean(cfg.getPermitAllAETitles()));
258                String[] caet = cfg.getCAET();
259                DataTable clientPermitACL = new DataTable(1, caet.length);
260                clientPermitACL.setColumnName(0, "Client AE Title");
261                if (caet.length < 1)
262                {
263                        clientPermitACL.addRow();
264                        clientPermitACL.setCellData(0, 0, "");
265                }
266                else
267                {
268                        for (int i = 0; i < caet.length; i++)
269                                clientPermitACL.setCellData(i, 0, caet[i]);
270                }
271                settings.put(SETTING_CLIENT_PERMIT_ACCESS_LIST_NAME, clientPermitACL);
272
273                return settings;
274        }

312        public boolean setAccessListSettings(HashMap<String, Object> settings)
313        {
314                // validate the settings
315                boolean valid = false;
316                try
317                {
318                        valid = tryAccessListSettings(settings);
319                }
320                catch (InvalidSettingValueException ex)
321                {
322                        // at least one of the setting is invalid, abort
323                        return false;
324                }
325                if (! valid) //settings not valid due to unknown reason, also abort
326                        return false;
327
328                // all the settings were correctly validated, so apply them all
329                cfg.setAE((String) settings.get(SETTING_SERVER_AE_TITLE_NAME));
330                cfg.setPermitAllAETitles(((Boolean) settings.get(SETTING_PERMIT_ALL_AE_TITLES_NAME)).booleanValue());
331                DataTable clientPermitACL = (DataTable) settings.get(SETTING_CLIENT_PERMIT_ACCESS_LIST_NAME);
332                LinkedList<String> caet = new LinkedList<String>();
333                if ((clientPermitACL.getRowCount() != 1) || (clientPermitACL.getCellData(0, 0) != ""))
334                        for (int i = 0; i < clientPermitACL.getRowCount(); i++)
335                                caet.add((String) clientPermitACL.getCellData(i, 0));
336                cfg.setCAET(caet.toArray(new String[caet.size()]));
337
338                // return success
339                return true;
340        }

398         *
399         * @param settings a Map containing the new settings.
400         * @return true if all settings were successfully changed to the ones supplied, false otherwise (no settings changed).
401         */
402        public boolean setSOPClassGlobalSettings(HashMap<String, Object> settings)
403        {
404                // validate the settings
405                boolean valid = false;
406                try
407                {
408                        valid = tryAccessListSettings(settings);
409                }
410                catch (InvalidSettingValueException ex)
411                {
412                        // at least one of the setting is invalid, abort
413                        return false;
414                }
415                if (! valid) //settings not valid due to unknown reason, also abort
416                        return false;
417
418                // all the settings were correctly validated, so apply them all
419                ComboBox tristate = (ComboBox) settings.get(SETTING_SOP_CLASS_ACCEPT_ALL_NAME);
420                Boolean accepted = null;
421                if (! tristate.getCurrentValue().isEmpty()) // NOTE: empty ("") means no change, see the getter for this settings
422                        accepted = new Boolean(parseCheckBoxValue(tristate.getCurrentValue()));
423                StaticDataTable table = (StaticDataTable) settings.get(SETTING_SOP_CLASS_GLOBAL_TRANSFER_STORAGE_NAME);
424
425                setTransferSettingsForAllSOPClasses(accepted, table);
426
427                // return success
428                return true;
429        }

673                 * @param trans the TransferStorage object.
674                 * @param accepted if this SOP Class is accepted. Can be null and if so, the target accepted value will remain the same.
675                 * @param allowedTransStore a HashMap indicating which TransferStore UIDs are accepted.
676                 */
677                public synchronized void setTransferStorageSettings(TransfersStorage trans, Boolean accepted, HashMap<String, Boolean> allowedTransStore)
678                {
679                        // accept this sop class, if necessary
680                        if (accepted != null)
681                                trans.setAccepted(accepted.booleanValue());
682                        // accept all the transfer storage settings
683                        for (String transStoreUID : transferSettings.keySet())
684                        {
685                                // skip this transfer store uid if it is not referenced on the allowedTransStore // FIXME is this right, or should non-defined entries be set as false?!?
686                                if (! allowedTransStore.containsKey(transStoreUID))
687                                        continue;
688
689                                trans.setTS(allowedTransStore.get(transStoreUID).booleanValue(), transferSettingsIndex.get(transStoreUID).intValue());
690                        }
691                }

694        /**
695         * Returns a StaticDataTable containing the TransferStorages for a
696         * specific SOP Class UID.
697         *
698         * @param sopClassUID a SOP Class UID.
699         * @return an Object array containing as first elements a Boolean object
700         * indicating if this SOP Class is accepted and as second element a
701         * StaticDataTable that contains the TransferStorages for this specific
702         * SOP Class UID.
703         */
704        public Object[] getTransferSettingsForSOPClass(String sopClassUID)
705        {
706                SOPClassSettings sops = SOPClassSettings.getInstance();
707                TransfersStorage ts = sops.getSOPClassSettings(sopClassUID);
708
709                // this settings will be returned on a StaticDataTable, to guarantee that the user does not change the entry count and only their values
710                StaticDataTable table = new StaticDataTable(1, sops.getTransferSettings().size());
711                // for each setting on the transfer storage settings...
712                int i = 0;
713                for (Map.Entry<String, String> entry : sops.getTransferSettings().entrySet())
714                {
715                        // get its UID, "visual/regular" name and index (relative to the TS settings array)
716                        String uid = entry.getKey();
717                        String name = entry.getValue();
718                        int index = sops.getTransferSettingsIndex().get(uid).intValue();
719
720                        // create a HintedCheckbox with the info gotten above and the current value of the setting
721                        CheckboxWithHint box = new CheckboxWithHint(ts.getTS()[index], name, uid);
722                        // and add the checkbox to the data table
723                        table.setCellData(i, 0, box);
724
725                        i++;
726                }
727
728                // mount the return object array
729                Object[] result = new Object[2];
730                // and add the proper objects to it
731                result[0] = new Boolean(ts.getAccepted());
732                result[1] = table;
733
734                return result;
735        }

784        /**
785         * Applies the Transfer Storage settings present on a StaticDataTable to
786         * a specific SOP Class UID.
787         *
788         * @param sopClassUID the target SOP Class UID.
789         * @param accepted if the SOP Class UID is accepted. Can be null and if so, the target accepted value will remain the same.
790         * @param table a StaticDataTable with the Transfer Storage settings.
791         */
792        public void setTransferSettingsForSOPClass(String sopClassUID, Boolean accepted, StaticDataTable table)
793        {
794                SOPClassSettings sops = SOPClassSettings.getInstance();
795                TransfersStorage ts = sops.getSOPClassSettings(sopClassUID);
796
797                // for each transfer storage setting on the datatable...
798                for (int i = 0 ; i < table.getRowCount(); i++)
799                {
800                        // get the current row/cell setting
801                        CheckboxWithHint box = (CheckboxWithHint) table.getCellData(i, 0);
802
803                        // get its UID and index (relative to the TS settings array)
804                        String uid = box.getHint();
805                        int index = sops.getTransferSettingsIndex().get(uid).intValue();
806
807                        // set the transfer storge setting value
808                        ts.setTS(box.isChecked(), index);
809                        if (accepted != null)
810                                ts.setAccepted(accepted.booleanValue());
811                }
812        }
