019package pt.ua.dicoogle.sdk.settings.types;
020
021import java.util.HashMap;
022import pt.ua.dicoogle.sdk.settings.Utils;
023
024/**
025 * Handles a data (type) that is suposed to be on a table form.
026 * Supports multiple fields per row.
027 * The contents of this table cannot be modified by the end user on the HTML interface.
028 *
029 * @author Ant√≥nio Novo <antonio.novo@ua.pt>
030 */
031public class StaticDataTable extends DataTable
032{
033        /**
034         * Creates a DataTable with no columns nor cells.
035         */
036        public StaticDataTable()
037        {
038                super();
039        }
040
041        /**
042         * Creates a DataTable with (unamed) columnCount columns.
043         *
044         * @param columnCount the initial number of columns.
045         */
046        public StaticDataTable(int columnCount)
047        {
048                super(columnCount);
049        }
050
051        /**
052         * Creates a DataTable with (unamed) columnCount columns and (empty) rowCount rows.
053         *
054         * @param columnCount the initial number of columns.
055         * @param rowCount the initial number of rows.
056         */
057        public StaticDataTable(int columnCount, int rowCount)
058        {
059                super(columnCount, rowCount);
060        }
061
062        // Output formats ------------------------------------------------------
063
064        @Override
065        public synchronized String toHTMLString(String htmlElementID)
066        {
067                String result = "";
068
069                result += "<div id=\"" + htmlElementID + "\" class=\"data-table\">";
070
071                // loop through all the rows and add their cell data and column name to the result
072                for (int row = 0; row < this.getRowCount(); row++)
073                {
074                        result += "<div class=\"data-table-row\">";
075
076                        for (int column = 0; column < this.getColumnCount(); column++)
077                        {
078                                String cellID = Utils.getHTMLElementIDFromString(htmlElementID + this.getColumnName(column) + row);
079
080                                result +=               Utils.getHTMLInputFromType(cellID, this.getCellData(row, column), false);
081                        }
082
083                        result += "</div>";
084                }
085
086                result += "</div>";
087
088                return result;
089        }
090
091        // Input formats -------------------------------------------------------
092
093        /**
094         * Based on a list of http request form (names & values) and a original value object,
095         * parse the first ones to match the second ones format with the new values.
096         *
097         * @param original the original DataTable.
098         * @param params the new params, and their values, to put into the DataTable.
099         * @return a new DataTable with the updated cell information.
100         */
101        @Override
102        public StaticDataTable fromHTTPParams(HashMap<String, String[]> params, int index, String htmlElementID)
103        {
104                // create a new DataTable with the same column size as the original one and the same row count as the new params
105                StaticDataTable result = new StaticDataTable(this.getColumnCount(), this.getRowCount());
106
107                // add all the columns titles/names
108                for (int i = 0; i < result.getColumnCount(); i++)
109                        result.setColumnName(i, this.getColumnName(i));
110
111                // add all the cells information available on the original DataTable
112                for (int column = 0; column < result.getColumnCount(); column++)
113                {
114                        for (int row = 0; row < result.getRowCount(); row++)
115                        {
116                                String cellID = Utils.getHTMLElementIDFromString(htmlElementID + result.getColumnName(column)) + row;
117                                Object cellData = this.getCellData(row, column);
118
119                                cellData = Utils.convertStringValueIntoProperType(cellData, params, 0, cellID);
120
121                                result.setCellData(row, column, cellData);
122                        }
123                }
124
125                // return the newly formed DataTable
126                return result;
127        }
128}
