208    public void testSave() throws IOException {
209        Path target = Files.createTempFile("conf", ".xml");
210
211        // Create default settings
212        ServerSettings settings = ServerSettingsImpl.createDefault();
213        final ServerSettings.DicomServices dcm = settings.getDicomServicesSettings();
214
215        settings.getArchiveSettings().setNodeName("dicoogle17");
216        settings.getArchiveSettings().setDefaultStorage(Arrays.asList("filestorage", "dropbox"));
217        MoveDestination md = new MoveDestination("MORE-TESTING", "192.168.0.55",
218                6060, true, "DESCRIPTION");
219        dcm.addMoveDestination(md);
220        dcm.getStorageSettings().setPort(6767);
221
222        ServerSettingsManager.saveSettingsTo(settings, target);
223
224        // read from file
225        settings = ServerSettingsManager.loadSettingsAt(target);
226        assertEquals("dicoogle17", settings.getArchiveSettings().getNodeName());
227        assertSameContent(Arrays.asList("filestorage", "dropbox"),
228                settings.getArchiveSettings().getDefaultStorage());
229        assertEquals(6767, dcm.getStorageSettings().getPort());
230
231        Collection<MoveDestination> destinations = Collections.singleton(md);
232        assertSameContent(destinations, dcm.getMoveDestinations());
233
234        // default
235        Collection<String> defaultTS = Arrays.asList(
236                "1.2.840.10008.1.2", "1.2.840.10008.1.2.1", "1.2.840.10008.1.2.2",
237                "1.2.840.10008.1.2.4.70", "1.2.840.10008.1.2.4.80", "1.2.840.10008.1.2.4.57",
238                "1.2.840.10008.1.2.4.90", "1.2.840.10008.1.2.4.91", "1.2.840.10008.1.2.5",
239                "1.2.840.10008.1.2.4.100", "1.2.840.10008.1.2.4.51");
240        assertSameContent(defaultTS, ((DicomServicesImpl)dcm).getDefaultTransferSyntaxes());
241
242        Collection<SOPClass> expectedRawSOP = SOPList.getInstance().asSOPClassList();
243        assertSameContent(expectedRawSOP, ((DicomServicesImpl)dcm).getRawSOPClasses());
244
245        // test remove move destination
246        boolean removed = settings.getDicomServicesSettings().removeMoveDestination(md.getAETitle());
247        assertTrue(removed);
248        assertTrue(settings.getDicomServicesSettings().getMoveDestinations().isEmpty());
249
250        // clean up
251        Files.delete(target);
252    }

359    private static void assertSameContent(Collection<?> o1, Collection<?> o2) {
360        assertNotNull("left-hand collection is null", o1);
361        assertNotNull("right-hand collection is null", o2);
362        assertEquals("Collection sizes do not match", o1.size(), o2.size());
363        for (Object o : o1) {
364            if (!o2.contains(o)) {
365                throw new ComparisonFailure("collection contents do not match", String.valueOf(o1), String.valueOf(o2));
366            }
367        }
368        for (Object o : o2) {
369            if (!o1.contains(o)) {
370                throw new ComparisonFailure("collection contents do not match", String.valueOf(o1), String.valueOf(o2));
371            }
372        }
373    }
