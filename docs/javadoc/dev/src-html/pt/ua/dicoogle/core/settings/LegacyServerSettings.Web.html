824    public boolean removeMoveDestination(String AETitle)
825    {
826        boolean removed = false;
827        Iterator<MoveDestination> it = dest.iterator();
828        while (it.hasNext()) {
829                MoveDestination mv = it.next();
830                if(mv.getAETitle().equals(AETitle))
831                {
832                        it.remove();
833                        removed = true;
834                }
835                        
836        }
837        return removed;
838    }

1164    public ArrayList<String> getNetworkInterfacesNames()
1165    {
1166        ArrayList<String> interfaces = new ArrayList<String>();
1167        Enumeration<NetworkInterface> nets = null;
1168        try
1169        {
1170            nets = NetworkInterface.getNetworkInterfaces();
1171        } catch (SocketException ex)
1172        {
1173            ex.printStackTrace();
1174        }
1175
1176        
1177        for (NetworkInterface netint : Collections.list(nets))
1178        {
1179            try
1180            {
1181                if (!netint.isLoopback())
1182                {
1183                    Enumeration<InetAddress> addresses = netint.getInetAddresses();
1184                    while (addresses.hasMoreElements())
1185                    {
1186                        if (Inet4Address.class.isInstance(addresses.nextElement()))
1187                        {
1188                            interfaces.add(netint.getDisplayName());
1189                        }
1190                    }
1191                }
1192            } catch (SocketException ex)
1193            {
1194                LoggerFactory.getLogger(LegacyServerSettings.class).error(ex.getMessage(), ex);
1195            }
1196        }
1197        return interfaces;
1198    }
