019package pt.ua.dicoogle.core.settings.part;
020
021import com.fasterxml.jackson.annotation.JsonAutoDetect;
022import com.fasterxml.jackson.annotation.JsonProperty;
023import com.fasterxml.jackson.dataformat.xml.annotation.JacksonXmlElementWrapper;
024import pt.ua.dicoogle.sdk.settings.server.ServerSettings;
025
026import java.util.Collections;
027import java.util.List;
028
029/**
030 * @author Eduardo Pinho <eduardopinho@ua.pt>
031 */
032@JsonAutoDetect(fieldVisibility = JsonAutoDetect.Visibility.ANY, getterVisibility = JsonAutoDetect.Visibility.NONE)
033public class ArchiveImpl implements ServerSettings.Archive {
034
035    /**
036     * Create the default server archive configurations
037     */
038    public static ArchiveImpl createDefault() {
039        ArchiveImpl a = new ArchiveImpl();
040
041        a.saveThumbnails = false;
042        a.thumbnailSize = 64;
043        a.indexerEffort = 100;
044        a.dirWatcherEnabled = false;
045        a.watchDirectory = "";
046
047        a.dimProviders = Collections.EMPTY_LIST;
048        a.defaultStorage = Collections.EMPTY_LIST;
049
050        return a;
051    }
052
053    @JsonProperty("save-thumbnails")
054    private boolean saveThumbnails;
055
056    @JsonProperty("thumbnail-size")
057    private int thumbnailSize;
058
059    @JsonProperty("indexer-effort")
060    private int indexerEffort;
061
062    @JsonProperty("dim-provider")
063    @JacksonXmlElementWrapper(localName = "dim-providers")
064    private List<String> dimProviders;
065
066    @JsonProperty("default-storage")
067    @JacksonXmlElementWrapper(localName = "default-storages")
068    private List<String> defaultStorage;
069
070    @JsonProperty(value = "enable-watch-directory", defaultValue = "false")
071    private boolean dirWatcherEnabled;
072
073    @JsonProperty("watch-directory")
074    private String watchDirectory;
075
076    @Override
077    public String getMainDirectory() {
078        return mainDirectory;
079    }
080
081    @Override
082    public void setMainDirectory(String mainDirectory) {
083        this.mainDirectory = mainDirectory;
084    }
085
086    @JsonProperty("main-directory")
087    private String mainDirectory;
088
089    @JsonProperty("node-name")
090    private String nodeName;
091
092    public boolean getSaveThumbnails() {
093        return saveThumbnails;
094    }
095
096    public void setSaveThumbnails(boolean saveThumbnails) {
097        this.saveThumbnails = saveThumbnails;
098    }
099
100    public int getThumbnailSize() {
101        return thumbnailSize;
102    }
103
104    public void setThumbnailSize(int thumbnailSize) {
105        this.thumbnailSize = thumbnailSize;
106    }
107
108    public int getIndexerEffort() {
109        return indexerEffort;
110    }
111
112    public void setIndexerEffort(int indexerEffort) {
113        this.indexerEffort = indexerEffort;
114    }
115
116    public List<String> getDIMProviders() {
117        return dimProviders;
118    }
119
120    public void setDIMProviders(List<String> dimProviders) {
121        this.dimProviders = dimProviders;
122    }
123
124    public List<String> getDefaultStorage() {
125        return defaultStorage;
126    }
127
128    public void setDefaultStorage(List<String> defaultStorage) {
129        this.defaultStorage = defaultStorage;
130    }
131
132    public boolean isDirectoryWatcherEnabled() {
133        return dirWatcherEnabled;
134    }
135
136    public void setDirectoryWatcherEnabled(boolean dirWatcherEnabled) {
137        this.dirWatcherEnabled = dirWatcherEnabled;
138    }
139
140    public String getWatchDirectory() {
141        return watchDirectory;
142    }
143
144    public void setWatchDirectory(String watchDirectory) {
145        this.watchDirectory = watchDirectory;
146    }
147
148    @Override
149    public String getNodeName() {
150        return nodeName;
151    }
152
153    @Override
154    public void setNodeName(String nodeName) {
155        this.nodeName = nodeName;
156    }
157
158    @Override
159    public String toString() {
160        return "ArchiveImpl{" +
161                "saveThumbnails=" + saveThumbnails +
162                ", thumbnailSize=" + thumbnailSize +
163                ", indexerEffort=" + indexerEffort +
164                ", dimProviders=" + dimProviders +
165                ", defaultStorage=" + defaultStorage +
166                ", dirWatcherEnabled=" + dirWatcherEnabled +
167                ", watchDirectory='" + watchDirectory + '\'' +
168                ", mainDirectory='" + mainDirectory + '\'' +
169                ", nodeName='" + nodeName + '\'' +
170                '}';
171    }
172}
