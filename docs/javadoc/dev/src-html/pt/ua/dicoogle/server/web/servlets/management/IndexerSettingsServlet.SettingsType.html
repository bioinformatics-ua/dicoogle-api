020package pt.ua.dicoogle.server.web.servlets.management;
021
022import java.io.IOException;
023
024import javax.servlet.ServletException;
025import javax.servlet.http.HttpServlet;
026import javax.servlet.http.HttpServletRequest;
027import javax.servlet.http.HttpServletResponse;
028
029import net.sf.json.JSONObject;
030
031import org.apache.commons.lang3.StringUtils;
032
033import pt.ua.dicoogle.core.settings.ServerSettingsManager;
034
035/**
036 *
037 * @author Frederico Silva <fredericosilva@ua.pt>
038 */
039public class IndexerSettingsServlet extends HttpServlet {
040
041    public enum SettingsType {
042
043        all,path, zip, effort, thumbnail,thumbnailSize, watcher;
044    }
045    private final SettingsType type;
046
047    public IndexerSettingsServlet(SettingsType type) {
048        this.type = type;
049    }
050
051    @Override
052    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
053        
054        String param = null;
055        String path = null;
056        boolean watcher=false, saveT = false;
057        int ieffort = 0;
058        int thumbnailSize = -1;
059        if(type != SettingsType.all)
060        {
061                param = req.getParameter(type.toString());
062        
063                if (StringUtils.isEmpty(param)) {
064                resp.sendError(400, "Invalid " + type.toString());
065            }
066        }
067        else{
068                path = req.getParameter("path");
069                watcher = Boolean.parseBoolean(req.getParameter("watcher"));
070                saveT = Boolean.parseBoolean(req.getParameter("saveThumbnail"));
071                ieffort = Integer.parseInt(req.getParameter("effort"));
072                thumbnailSize = Integer.parseInt(req.getParameter("thumbnailSize"));
073        }
074
075        switch (type) {
076            case path:
077                ServerSettingsManager.getSettings().getArchiveSettings().setWatchDirectory(param);
078                resp.getWriter().append("Dir set to: " + param);
079                break;
080            case zip:
081                resp.getWriter().append("indexZipFiles is no longer supported");
082                break;
083            case effort:
084                //TODO: CHECK ACCPTABLE RANGE
085                int effort = Integer.parseInt(param);
086                ServerSettingsManager.getSettings().getArchiveSettings().setIndexerEffort(effort);
087                resp.getWriter().append("index effort set to: " + effort);
088                break;
089            case thumbnail:
090                boolean saveThumbanail = Boolean.parseBoolean(param);
091                ServerSettingsManager.getSettings().getArchiveSettings().setSaveThumbnails(saveThumbanail);
092                break;
093            case thumbnailSize:
094                //TODO: Should be a int
095                //int thumbSize = Integer.parseInt(param);
096                ServerSettingsManager.getSettings().getArchiveSettings().setThumbnailSize(Integer.parseInt(param));
097                break;
098            case watcher:
099                ServerSettingsManager.getSettings().getArchiveSettings().setDirectoryWatcherEnabled(Boolean.parseBoolean(param));
100                break;
101            case all:
102                 ServerSettingsManager.getSettings().getArchiveSettings().setWatchDirectory(path);
103                 if (ieffort > 0 && ieffort <= 100) {
104                     ServerSettingsManager.getSettings().getArchiveSettings().setIndexerEffort(ieffort);
105                 }
106                 ServerSettingsManager.getSettings().getArchiveSettings().setSaveThumbnails(saveT);
107                 if (thumbnailSize > 0) {
108                     ServerSettingsManager.getSettings().getArchiveSettings().setThumbnailSize(thumbnailSize);
109                 }
110                 ServerSettingsManager.getSettings().getArchiveSettings().setDirectoryWatcherEnabled(watcher);
111                break;
112                
113        }
114        ServerSettingsManager.saveSettings();
115    }
116
117    @Override
118    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
119        String result = "";
120        resp.setContentType("application/json");
121        switch (type) {
122            case path:
123                result = ServerSettingsManager.getSettings().getArchiveSettings().getMainDirectory();
124                break;
125            case zip:
126                result = "false"; // String.valueOf(ServerSettingsManager.getSettings().isGzipStorage());
127                break;
128            case effort:
129                result = String.valueOf(ServerSettingsManager.getSettings().getArchiveSettings().getIndexerEffort());
130                break;
131            case thumbnail:
132                result = String.valueOf(ServerSettingsManager.getSettings().getArchiveSettings().getSaveThumbnails());
133                break;
134            case thumbnailSize:
135                result = String.valueOf(ServerSettingsManager.getSettings().getArchiveSettings().getThumbnailSize());
136                break;
137            case watcher:
138                result = String.valueOf(ServerSettingsManager.getSettings().getArchiveSettings().isDirectoryWatcherEnabled());
139                break;
140            case all:
141                JSONObject allresponse = new JSONObject();
142                allresponse.put("path", ServerSettingsManager.getSettings().getArchiveSettings().getMainDirectory());
143                allresponse.put("zip", "false"); // ServerSettingsManager.getSettings().isGzipStorage());
144                allresponse.put("effort", String.valueOf(ServerSettingsManager.getSettings().getArchiveSettings().getIndexerEffort()));
145                allresponse.put("thumbnail", ServerSettingsManager.getSettings().getArchiveSettings().getSaveThumbnails());
146                allresponse.put("thumbnailSize", String.valueOf(ServerSettingsManager.getSettings().getArchiveSettings().getThumbnailSize()));
147                allresponse.put("watcher", ServerSettingsManager.getSettings().getArchiveSettings().isDirectoryWatcherEnabled());
148                
149                resp.getWriter().write(allresponse.toString());
150                break;
151        }
152
153        resp.setContentType("application/json");
154        resp.getWriter().append(result);
155    }
156
157}
