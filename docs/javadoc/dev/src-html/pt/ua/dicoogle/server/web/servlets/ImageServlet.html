057/**
058 * Handles the requests for DICOM frames, returning them as PNG images.
059 * Also maintains a cache of the images already served to speed-up the next requests (minimizing server load by doing way less conversions).
060 *
061 * @author Antonio
062 * @author Eduardo Pinho <eduardopinho@ua.pt>
063 */
064public class ImageServlet extends HttpServlet
065{
066    private static final Logger logger = LoggerFactory.getLogger(ImageServlet.class);
067    private static final long serialVersionUID = 1L;
068
069    public static final int BUFFER_SIZE = 1500; // byte size for read-write ring bufer, optimized for regular TCP connection windows
070
071    private final LocalImageCache cache;
072
073    /**
074     * Creates an image servlet.
075     *
076     * @param cache the local image caching system, can be null and if so no caching mechanism will be used.
077     */
078    public ImageServlet(LocalImageCache cache) {
079        this.cache = cache;
080    }
081    
082    @Override
083    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException
084    {
085        String sopInstanceUID = request.getParameter("SOPInstanceUID");
086        String uri = request.getParameter("uri");
087        String qsThumbnail = request.getParameter("thumbnail");
088        boolean thumbnail = qsThumbnail != null && (qsThumbnail.isEmpty() || Boolean.valueOf(qsThumbnail));
089
090        if (sopInstanceUID == null) {
091            if (uri == null) {
092                response.sendError(400, "URI or SOP Instance UID not provided");
093                return;
094            }
095        } else {
096            sopInstanceUID = sopInstanceUID.trim();
097            if (sopInstanceUID.isEmpty()) {
098                response.sendError(400, "Invalid SOP Instance UID!");
099                return;
100            }
101        }
102        String[] providerArray = request.getParameterValues("provider");
103        List<String> providers = providerArray == null ? null : Arrays.asList(providerArray);
104        String sFrame = request.getParameter("frame");
105        int frame;
106        if (sFrame == null) {
107            frame = 0;
108        } else {
109            frame = Integer.parseInt(sFrame);
110        }
111        
112        StorageInputStream imgFile;
113        if (sopInstanceUID != null) {
114            try {
115                // get the image file for that SOP Instance UID
116                imgFile = getFileFromSOPInstanceUID(sopInstanceUID, providers);
117            } catch (QueryParseException ex) {
118                ResponseUtil.sendError(response, 400, "Failed to parse the query: " + ex.getMessage());
119                return;
120            }
121            // if no .dcm file was found tell the client
122            if (imgFile == null) {
123                ResponseUtil.sendError(response, 404, "No image file for supplied SOP Instance UID");
124                return;
125            }
126        } else {
127            try {
128                // get the image file by the URI
129                URI imgUri = new URI(uri);
130                StorageInterface storageInt = PluginController.getInstance().getStorageForSchema(imgUri);
131                Iterator<StorageInputStream> storages = storageInt.at(imgUri).iterator();
132                // take the first valid storage
133                if (!storages.hasNext()) {
134                    ResponseUtil.sendError(response, 404, "No image file for supplied URI");
135                    return;
136                }
137                imgFile = storages.next();
138                 
139            } catch (URISyntaxException ex) {
140                ResponseUtil.sendError(response, 400, "Bad URI syntax");
141                return;
142            }
143        }
144
145        // if there is a cache available then use it
146        if (cache != null && cache.isRunning()) {
147
148            try {
149                InputStream istream = cache.get(imgFile.getURI(), frame, thumbnail);
150                response.setContentType("image/png");
151                try(ServletOutputStream out = response.getOutputStream()) {
152                    IOUtils.copy(istream, out);
153                }
154            } catch (IOException ex) {
155                logger.warn("Could not convert the image", ex);
156                ResponseUtil.sendError(response, 500, "Failed to convert the image");
157            } catch (RuntimeException ex) {
158                logger.error("Unexpected exception", ex);
159                ResponseUtil.sendError(response, 500, "Unexpected internal server error");
160            }
161            
162        } else {
163            // if the cache is invalid or not running convert the image and return it "on-the-fly"
164            try {
165                ByteArrayOutputStream pngStream = getPNGStream(imgFile, frame, thumbnail);
166                response.setContentType("image/png"); // set the appropriate type for the PNG image
167                response.setContentLength(pngStream.size()); // set the image size
168                try (ServletOutputStream out = response.getOutputStream()) {
169                    pngStream.writeTo(out);
170                    pngStream.flush();
171                }
172            } catch (IOException ex) {
173                logger.warn("Could not convert the image", ex);
174                ResponseUtil.sendError(response, 500, "Failed to convert the image");
175            }
176        }
177    }
178    
179    private ByteArrayOutputStream getPNGStream(StorageInputStream imgFile, int frame, boolean thumbnail) throws IOException {
180        ByteArrayOutputStream pngStream;
181        if (thumbnail) {
182            // retrieve thumbnail dimension settings
183            int thumbSize = ServerSettingsManager.getSettings().getArchiveSettings().getThumbnailSize();
184            pngStream = Convert2PNG.DICOM2ScaledPNGStream(imgFile, frame, thumbSize, thumbSize);
185        } else {
186            pngStream = Convert2PNG.DICOM2PNGStream(imgFile, frame);
187        }
188        return pngStream;
189    }

191        @Override
192        protected void doPost(HttpServletRequest req, HttpServletResponse resp)
193                        throws ServletException, IOException {
194        // TODO this service does not make sense as a POST.
195        // either remove or relocate to another resource
196        
197                String sop = req.getParameter("SOPInstanceUID");
198                if(sop == null){
199                        resp.sendError(500, "No SOPInstanceUID in Request");
200                        return;
201                }
202
203                float frameRate = Information.getFrameRateFromImage(sop);
204                if(frameRate == -1){
205                        resp.sendError(500, "Cannot Locate the image File.");
206                        return;
207                }
208                
209                int nFrames = Information.getNumberOfFramesInFile(sop);
210                
211                JSONObject r = new JSONObject();
212                r.put("SOPInstanceUID", sop);
213                r.put("NumberOfFrames", nFrames);
214                r.put("FrameRate", frameRate);
215                
216                resp.setContentType("application/json");
217                
218                PrintWriter wr = resp.getWriter();
219                wr.print(r.toString()); 
220        }

222    private static StorageInputStream getFileFromSOPInstanceUID(String sopInstanceUID, List<String> providers) throws IOException {
223        // TODO use only DIM sources?
224        JointQueryTask qt = new JointQueryTask() {
225            @Override
226            public void onCompletion() {
227            }
228            @Override
229            public void onReceive(Task<Iterable<SearchResult>> e) {
230            }
231        };
232        try {
233            if (providers == null) {
234                providers = PluginController.getInstance().getQueryProvidersName(true);
235            }
236            Iterator<SearchResult> it = PluginController.getInstance()
237                    .query(qt, providers, "SOPInstanceUID:" + sopInstanceUID).get().iterator();
238            if (!it.hasNext()) {
239                throw new IOException("No such image of SOPInstanceUID " + sopInstanceUID);
240            }
241            SearchResult res = it.next();
242            StorageInterface storage = PluginController.getInstance().getStorageForSchema(res.getURI());
243            if (storage == null) {
244                throw new IOException("Unsupported file scheme");
245            }
246            Iterator<StorageInputStream> store = storage.at(res.getURI()).iterator();
247            if (!store.hasNext()) {
248                throw new IOException("No storage item found");
249            }
250            return store.next();
251        } catch (InterruptedException | ExecutionException ex) {
252            throw new IOException(ex);
253        }        
254    }
