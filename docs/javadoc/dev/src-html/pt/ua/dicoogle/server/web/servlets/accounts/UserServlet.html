020package pt.ua.dicoogle.server.web.servlets.accounts;
021
022import java.io.IOException;
023import java.util.Set;
024import java.util.regex.Matcher;
025import java.util.regex.Pattern;
026import javax.servlet.ServletException;
027import javax.servlet.http.HttpServlet;
028import javax.servlet.http.HttpServletRequest;
029import javax.servlet.http.HttpServletResponse;
030
031import net.sf.json.JSONArray;
032import net.sf.json.JSONObject;
033import pt.ua.dicoogle.server.users.HashService;
034import pt.ua.dicoogle.server.users.User;
035import pt.ua.dicoogle.server.users.UsersStruct;
036import pt.ua.dicoogle.server.web.utils.ResponseUtil;
037
038/**
039 * User Servlet for create, remove and consult user accounts
040 *
041 * @author Frederico Silva <fredericosilva@ua.pt>
042 */
043public class UserServlet extends HttpServlet {
044
045    @Override
046    protected void doDelete(HttpServletRequest req, HttpServletResponse resp) throws IOException {
047        String pathInfo = req.getPathInfo();
048
049        if (pathInfo == null) {
050            ResponseUtil.sendError(resp, HttpServletResponse.SC_BAD_REQUEST, "Username not provided");
051            return;
052        }
053
054        String[] params = pathInfo.replaceFirst("^/", "").split("/");
055
056        if (params.length < 1) {
057            ResponseUtil.sendError(resp, HttpServletResponse.SC_BAD_REQUEST, "Username not provided");
058            return;
059        }
060
061        String usernameToRemove = params[0];
062        boolean isRemoved = false;
063        if (usernameToRemove != null && !usernameToRemove.equals("")) {
064            isRemoved = UsersStruct.getInstance().removeUser(usernameToRemove);
065        }
066
067        ResponseUtil.simpleResponse(resp, "success", isRemoved);
068    }
069
070    @Override
071    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws IOException {
072        String user = req.getParameter("username");
073        String pass = req.getParameter("password");
074        boolean admin = Boolean.parseBoolean(req.getParameter("admin"));
075
076        String passHash = HashService.getSHA1Hash(pass);             //password Hash
077        String hash = HashService.getSHA1Hash(user + admin + passHash);   //user Hash
078
079        boolean wasAdded = UsersStruct.getInstance().addUser(new User(user, hash, admin));
080
081
082        ResponseUtil.simpleResponse(resp, "success", wasAdded);
083    }
084
085    @Override
086    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
087
088        Set<String> users = UsersStruct.getInstance().getUsernames();
089        resp.setContentType("application/json");
090
091        JSONObject jsonObject = new JSONObject();
092        JSONArray usersArray = new JSONArray();
093        for (String user : users) {
094            JSONObject u = new JSONObject();
095            u.put("username", user);
096            usersArray.add(u);
097        }
098
099        jsonObject.put("users", usersArray);
100        jsonObject.write(resp.getWriter());
101    }
102
103}
