019package pt.ua.dicoogle.server.users;
020
021import java.util.*;
022
023/**
024 * This class stores the list of users of Dicoogle
025 *
026 * @author Samuel Campos <samuelcampos@ua.pt>
027 * @author Rui Lebre <ruilebre@ua.pt>
028 */
029public class UsersStruct {
030    private HashMap<String, User> users;
031
032    private static UsersStruct instance = null;
033
034    // count the number of administrators
035    private int numberOfAdmins;
036    private UsersXML usersXML;
037
038    public static synchronized UsersStruct getInstance() {
039        if (instance == null) {
040            instance = new UsersStruct();
041        }
042
043        return instance;
044    }
045
046
047    private UsersStruct() {
048        reset();
049
050        usersXML = new UsersXML();
051        Collection<User> userList = usersXML.getXML();
052
053        for (User user : userList) {
054            users.put(user.getUsername(), user);
055            if(user.isAdmin()) {
056                numberOfAdmins++;
057            }
058        }
059
060    }
061
062
063    /**
064     * Insert one default user
065     * Username: "dicoogle"
066     * Password: "dicoogle" (hashed)
067     *
068     * This user is administrator
069     */
070    public static Collection<User> getDefaults() {
071        String username = "dicoogle";
072        boolean admin = true;
073        String passPlainText = "dicoogle";
074
075        String passHash = HashService.getSHA1Hash(passPlainText);             //password Hash
076        String hash = HashService.getSHA1Hash(username + admin + passHash);   //user Hash
077
078        return Collections.singleton(new User(username, hash, admin));
079    }
080
081    /**
082     * Used only by UsersXML to reset User Settings
083     */
084    protected void reset() {
085        users = new HashMap<>();
086        numberOfAdmins = 0;
087    }
088
089
090    /**
091     * Insert user in the List of users
092     *
093     * @param user
094     * @return true - if succeeded. false - if the username already exists
095     */
096    public boolean addUser(User user) {
097        if (users.containsKey(user.getUsername()))
098            return false;
099
100        users.put(user.getUsername(), user);
101
102        if (user.isAdmin())
103            numberOfAdmins++;
104
105        usersXML.printXML(this.getUsers());
106        return true;
107    }
108
109    /**
110     * Removes one user from de list
111     * Maintains at least one administrator in the list
112     * (refuses to remove the last one)
113     *
114     * @param username
115     * @return
116     */
117    public boolean removeUser(String username) {
118        if (username == null)
119            return false;
120
121        User user = users.get(username);
122        if (user == null)
123            return false;
124
125        if (user.isAdmin() && numberOfAdmins == 1)
126            return false;
127        else if (user.isAdmin())
128            numberOfAdmins--;
129
130        users.remove(username);
131        usersXML.printXML(users.values());
132
133        return true;
134    }
135
136    public Collection<User> getUsers() {
137        return users.values();
138    }
139
140    public Set<String> getUsernames() {
141        return users.keySet();
142    }
143
144    public User getUser(String username) {
145        return users.get(username);
146    }
147}
