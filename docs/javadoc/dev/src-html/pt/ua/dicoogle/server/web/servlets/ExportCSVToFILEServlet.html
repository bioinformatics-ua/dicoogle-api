021import java.io.BufferedInputStream;
022import java.io.BufferedOutputStream;
023import java.io.File;

025import java.io.FileOutputStream;
026import java.io.IOException;
027import java.util.*;
028import java.util.concurrent.atomic.AtomicLong;
029
030import javax.servlet.ServletException;
031import javax.servlet.http.HttpServlet;
032import javax.servlet.http.HttpServletRequest;
033import javax.servlet.http.HttpServletResponse;
034
035import org.apache.commons.io.IOUtils;
036
037import net.sf.json.JSONArray;
038import net.sf.json.JSONException;
039import net.sf.json.JSONObject;
040import org.slf4j.Logger;
041import org.slf4j.LoggerFactory;
042import pt.ua.dicoogle.core.QueryExpressionBuilder;
043import pt.ua.dicoogle.core.query.ExportToCSVQueryTask;
044import pt.ua.dicoogle.plugins.PluginController;
045import pt.ua.dicoogle.server.web.utils.ResponseUtil;
046
047public class ExportCSVToFILEServlet extends HttpServlet {
048        private static final Logger logger = LoggerFactory.getLogger(ExportCSVToFILEServlet.class);
049
050        private File tempDirectory;
051        public ExportCSVToFILEServlet(File tempDirectory) {
052                super();
053                this.tempDirectory = tempDirectory;
054        }
055
056        private static final long serialVersionUID = 1L;
057
058        @Override
059        protected void doGet(HttpServletRequest req, HttpServletResponse resp)
060                        throws ServletException, IOException {
061                String uid = req.getParameter("UID");
062                if(uid == null){
063                        ResponseUtil.sendError(resp, 400, "No query UID Supplied: Please provide the field \"UID\"");
064                        return;
065                }
066
067                // validate UUID
068                try {
069                        UUID.fromString(uid);
070                } catch (IllegalArgumentException ex) {
071                        ResponseUtil.sendError(resp, 400, "Illegal UUID supplied");
072                        return;
073                }       
074
075                File tmpFile = new File(tempDirectory, "QueryResultsExport-"+uid);
076                logger.debug("Reading temporary CSV file: {}", tmpFile);
077                if(!tmpFile.exists()) {
078                        ResponseUtil.sendError(resp, 404, "The file for the given UID was not found.");
079                        return; 
080                }
081                
082                // Find this file id in database to get file name, and file type
083
084        // You must tell the browser the file type you are going to send
085        // for example application/pdf, text/plain, text/html, image/jpg
086        resp.setContentType("application/csv");
087        // Make sure to show the download dialog
088        resp.setHeader("Content-disposition","attachment; filename=QueryResultsExport.csv");            
089                
090                try (BufferedInputStream bi = new BufferedInputStream(new FileInputStream(tmpFile))) {
091
092                        IOUtils.copy(bi, resp.getOutputStream());
093                        resp.getOutputStream().flush();
094                }
095        }
096
097        @Override
098        protected void doPost(HttpServletRequest req, HttpServletResponse resp)
099                        throws ServletException, IOException {
100                List<String> orderedFields = new ArrayList<>();
101                Map<String, String> fields = new HashMap<>();
102                String queryString = null;
103                String[] arr;           
104                boolean keyword;
105                
106                try {
107                        queryString = req.getParameter("query");
108                        if (queryString == null) {
109                                ResponseUtil.sendError(resp, 400,
110                                                "QueryString not supplied: Please fill the field \"query\"");
111                                return;
112                        }
113
114                        logger.debug("{}", req.getParameter("fields"));
115                        JSONArray jsonObj = JSONArray.fromObject(req.getParameter("fields"));
116                        if (jsonObj.size()== 0) {
117                                ResponseUtil.sendError(resp, 400,
118                                                "No fields supplied: Please fill the field \"extraFields\" in \"JSON\"");
119                                return;
120                        }
121
122                        for (Object f : jsonObj) {
123                                fields.put(f.toString(), f.toString());
124                                orderedFields.add(f.toString());
125                        }
126                        keyword = Boolean.parseBoolean(req.getParameter("keyword"));
127
128                        arr = req.getParameterValues("providers");
129                } catch (JSONException ex) {
130                        ResponseUtil.sendError(resp, 400, "Invalid JSON content");
131                        return;
132                }
133                
134                if (!keyword) {
135            QueryExpressionBuilder q = new QueryExpressionBuilder(queryString);
136            queryString = q.getQueryString();
137        }
138
139                String uid = UUID.randomUUID().toString();
140                //File tempFile = File.createTempFile("QueryResultsExport-", uid, tempDirectory);
141                File tempFile = new File(tempDirectory, "QueryResultsExport-"+uid);
142                tempFile.deleteOnExit();
143                logger.debug("UID: {}", uid);
144                logger.debug("FilePath: {}", tempFile.getAbsolutePath());
145                
146                FileOutputStream outStream = new FileOutputStream(tempFile);
147                BufferedOutputStream bos = new BufferedOutputStream(outStream);
148                
149                ExportToCSVQueryTask task = new ExportToCSVQueryTask(orderedFields,
150                                bos);
151                
152                if (arr == null || arr.length ==0) {
153                        PluginController.getInstance().queryAll(task, queryString, fields);
154                } else {
155                        List<String> providers = new ArrayList<>();
156                        for (Object f : arr) {
157                                providers.add(f.toString());
158                        }
159                        PluginController.getInstance().query(task, providers, queryString,
160                                        fields);
161                }
162
163                task.await();
164                
165                JSONObject obj = new JSONObject();
166                obj.put("uid", uid);
167                resp.getWriter().write(obj.toString());
168                resp.getWriter().flush();
169        }
170
171}
