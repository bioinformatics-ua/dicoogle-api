064        /*
065         * Action codes for internal use.
066         */
067        private static final int ACTION_CODE_INVALID = 0;
068        private static final int ACTION_CODE_SET_ADVANCED_SETTINGS = 1;
069        private static final int ACTION_CODE_SET_ACCESS_LIST_SETTINGS = 2;
070        private static final int ACTION_CODE_SET_SOP_CLASS_GLOBAL_TRANSFER_STORAGE_SETTINGS = 3;
071        private static final int ACTION_CODE_SET_STORAGE_SERVICES = 4;

073        @Override
074        protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException
075        {
076                // TODO validate user session credentials
077
078                // get which action to take
079                String actionStr = request.getParameter(PARAM_ACTION);
080
081                // translate each action string to a action identification
082                int action = ACTION_CODE_INVALID;
083                if (actionStr != null)
084                {
085                        if (actionStr.equalsIgnoreCase(ACTION_SET_ADVANCED_SETTINGS))
086                                action = ACTION_CODE_SET_ADVANCED_SETTINGS;
087                        else if (actionStr.equalsIgnoreCase(ACTION_SET_ACCESS_LIST_SETTINGS))
088                                action = ACTION_CODE_SET_ACCESS_LIST_SETTINGS;
089                        else if (actionStr.equalsIgnoreCase(ACTION_SET_SOP_CLASS_GLOBAL_TRANSFER_STORAGE_SETTINGS))
090                                action = ACTION_CODE_SET_SOP_CLASS_GLOBAL_TRANSFER_STORAGE_SETTINGS;
091                        else if (actionStr.equalsIgnoreCase(ACTION_SET_STORAGE_SERVICES))
092                                action = ACTION_CODE_SET_STORAGE_SERVICES;
093
094                }
095
096                // response to each action accordingly
097                switch (action)
098                {
099                        case ACTION_CODE_SET_ADVANCED_SETTINGS:
100                                HashMap<String, String[]> advSettings = new HashMap<>();
101
102                                // get all the settings and their values
103                                Enumeration<String> params = request.getParameterNames();
104                                while (params.hasMoreElements())
105                                {
106                                        String name = params.nextElement();
107
108                                        // ignore the main params (the ones that go us here)
109                                        if (name.equalsIgnoreCase(PARAM_ACTION))
110                                                continue;
111
112                                        String[] value = request.getParameterValues(name);
113                                        advSettings.put(name, value);
114                                }
115
116                                HashMap<String, Object> settings = getIndexingSettings();
117
118                                processAdvancedSettings(settings, advSettings);
119
120                                // try to apply the settings
121                                setIndexingSettings(settings);
122
123                                ServerSettingsManager.saveSettings();
124
125                                // send the client back the to previous page
126                                response.sendRedirect(Session.getLastVisitedURL(request));
127                        break;
128
129                        case ACTION_CODE_SET_ACCESS_LIST_SETTINGS:
130                                advSettings = new HashMap<String, String[]>();
131
132                                // get all the settings and their values
133                                params = request.getParameterNames();
134                                while (params.hasMoreElements())
135                                {
136                                        String name = params.nextElement();
137
138                                        // ignore the main params (the ones that go us here)
139                                        if (name.equalsIgnoreCase(PARAM_ACTION))
140                                                continue;
141
142                                        String[] value = request.getParameterValues(name);
143
144                                        advSettings.put(name, value);
145                                }
146
147                                settings = getAccessListSettings();
148
149                                processAdvancedSettings(settings, advSettings);
150
151                                // try to apply the settings
152                                setAccessListSettings(settings);
153
154                                ServerSettingsManager.saveSettings();
155                                // TODO force the reload of the IAccessList implementation for the RMI interface
156
157                                // send the client back the to previous page
158                                response.sendRedirect(Session.getLastVisitedURL(request));
159                        break;
160
161                        case ACTION_CODE_SET_SOP_CLASS_GLOBAL_TRANSFER_STORAGE_SETTINGS:
162                                advSettings = new HashMap<String, String[]>();
163
164                                // get all the settings and their values
165                                params = request.getParameterNames();
166                                while (params.hasMoreElements())
167                                {
168                                        String name = params.nextElement();
169
170                                        // ignore the main params (the ones that go us here)
171                                        if (name.equalsIgnoreCase(PARAM_ACTION))
172                                                continue;
173
174                                        String[] value = request.getParameterValues(name);
175
176                                        advSettings.put(name, value);
177                                }
178
179                                settings = getSOPClassGlobalSettings();
180
181                                processAdvancedSettings(settings, advSettings);
182
183                                setSOPClassGlobalSettings(settings);
184
185                                ServerSettingsManager.saveSettings();
186
187                                // send the client back the to previous page
188                                response.sendRedirect(Session.getLastVisitedURL(request));
189                        break;
190                        
191                        case ACTION_CODE_SET_STORAGE_SERVICES:
192
193                                String requestParam = request.getParameter("MOVES");
194
195                                if(requestParam == null){
196                                        response.sendError(401,"No Moves Defined");
197                                        return;
198                                }
199                                
200                                ServerSettings set = ServerSettingsManager.getSettings();
201                                ArrayList<MoveDestination> nmoves = new ArrayList<MoveDestination>();
202                                
203                                JSONArray moves = JSONArray.fromObject(requestParam);
204                                Iterator it = moves.iterator();
205                                while(it.hasNext()){
206                                        JSONObject obj = (JSONObject) it.next();
207                                        
208                                        try{
209                                                String aet = obj.getString("AETitle");
210                                                String ip = obj.getString("ipAddrs");
211                                                int port = obj.getInt("port");
212                                                
213                                                if( StringUtils.isNotBlank(aet) && StringUtils.isNotBlank(ip) && port > 0){
214                                                        nmoves.add(new MoveDestination(aet, ip, port)); 
215                                                }       
216                                        }catch(JSONException ex){
217                                                
218                                        }                       
219                                }
220                                
221                                set.getDicomServicesSettings().setMoveDestinations(nmoves);
222                                ServerSettingsManager.saveSettings();
223                                break;
224
225                        default:
226                                response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Invalid action!");
227                                return;
228                }
229        }

243        public void setIndexingSettings(HashMap<String, Object> settings)
244        {
245                ServerSettings.Archive cfg = ServerSettingsManager.getSettings().getArchiveSettings();
246
247                // all the settings were correctly validated, so apply them all
248                cfg.setWatchDirectory(((ServerDirectoryPath) settings.get(SETTING_DIRECTORY_MONITORIZATION_NAME)).getPath());
249                cfg.setDirectoryWatcherEnabled(((Boolean) settings.get(SETTING_DIRECTORY_WATCHER_NAME)).booleanValue());
250                cfg.setIndexerEffort(((RangeInteger) settings.get(SETTING_INDEXING_EFFORT_NAME)).getValue());
251                cfg.setThumbnailSize((Integer) settings.get(SETTING_THUMBNAILS_SIZE_NAME));
252                cfg.setSaveThumbnails(((Boolean) settings.get(SETTING_SAVE_THUMBNAILS_NAME)).booleanValue());
253                //cfg.setIndexZIPFiles(((Boolean) settings.get(SETTING_INDEX_ZIP_FILES_NAME)).booleanValue());
254        }

299         * Returns a Map containing the current settings.
300         *
301         * @return a Map containing the current settings.
302         */
303        public HashMap<String, Object> getAccessListSettings()
304        {
305                ServerSettings cfg = ServerSettingsManager.getSettings();
306
307                HashMap<String, Object> settings = new HashMap<>();
308
309                settings.put(SETTING_SERVER_AE_TITLE_NAME, cfg.getDicomServicesSettings().getAETitle());
310                settings.put(SETTING_PERMIT_ALL_AE_TITLES_NAME, new Boolean(cfg.getDicomServicesSettings().getAllowedAETitles().isEmpty()));
311                Collection<String> allowedAETitles = cfg.getDicomServicesSettings().getAllowedAETitles();
312                String[] caet = new String[allowedAETitles.size()];
313                allowedAETitles.toArray(caet);
314                DataTable clientPermitACL = new DataTable(1, caet.length);
315                clientPermitACL.setColumnName(0, "Client AE Title");
316                if (caet.length < 1)
317                {
318                        clientPermitACL.addRow();
319                        clientPermitACL.setCellData(0, 0, "");
320                }
321                else
322                {
323                        for (int i = 0; i < caet.length; i++)
324                                clientPermitACL.setCellData(i, 0, caet[i]);
325                }
326                settings.put(SETTING_CLIENT_PERMIT_ACCESS_LIST_NAME, clientPermitACL);
327
328                return settings;
329        }

374        {
375                // all the settings were correctly validated, so apply them all
376                ComboBox tristate = (ComboBox) settings.get(SETTING_SOP_CLASS_ACCEPT_ALL_NAME);
377                Boolean accepted = null;
378                if (! tristate.getCurrentValue().isEmpty()) // NOTE: empty ("") means no change, see the getter for this settings

380                StaticDataTable table = (StaticDataTable) settings.get(SETTING_SOP_CLASS_GLOBAL_TRANSFER_STORAGE_NAME);
381
382                setTransferSettingsForAllSOPClasses(accepted, table);
383
384                // return success
385                return true;

387        /**
388         * Returns a StaticDataTable containing the TransferStorages for a
389         * specific SOP Class UID.
390         *
391         * @param sopClassUID a SOP Class UID.
392         * @return an Object array containing as first elements a Boolean object
393         * indicating if this SOP Class is accepted and as second element a
394         * StaticDataTable that contains the TransferStorages for this specific
395         * SOP Class UID.
396         */
397        public Object[] getTransferSettingsForSOPClass(String sopClassUID)
398        {
399                SOPClassSettings sops = SOPClassSettings.getInstance();
400                TransfersStorage ts = sops.getSOPClassSettings(sopClassUID);
401
402                // this settings will be returned on a StaticDataTable, to guarantee that the user does not change the entry count and only their values
403                StaticDataTable table = new StaticDataTable(1, sops.getTransferSettings().size());
404                // for each setting on the transfer storage settings...
405                int i = 0;
406                for (Map.Entry<String, String> entry : sops.getTransferSettings().entrySet())
407                {
408                        // get its UID, "visual/regular" name and index (relative to the TS settings array)
409                        String uid = entry.getKey();
410                        String name = entry.getValue();
411                        int index = sops.getTransferSettingsIndex().get(uid).intValue();
412
413                        // create a HintedCheckbox with the info gotten above and the current value of the setting
414                        CheckboxWithHint box = new CheckboxWithHint(ts.getTS()[index], name, uid);
415                        // and add the checkbox to the data table
416                        table.setCellData(i, 0, box);
417
418                        i++;
419                }
420
421                // mount the return object array
422                Object[] result = new Object[2];
423                // and add the proper objects to it
424                result[0] = new Boolean(ts.getAccepted());
425                result[1] = table;
426
427                return result;
428        }

477        /**
478         * Applies the Transfer Storage settings present on a StaticDataTable to
479         * a specific SOP Class UID.
480         *
481         * @param sopClassUID the target SOP Class UID.
482         * @param accepted if the SOP Class UID is accepted. Can be null and if so, the target accepted value will remain the same.
483         * @param table a StaticDataTable with the Transfer Storage settings.
484         */
485        public void setTransferSettingsForSOPClass(String sopClassUID, Boolean accepted, StaticDataTable table)
486        {
487                SOPClassSettings sops = SOPClassSettings.getInstance();
488                TransfersStorage ts = sops.getSOPClassSettings(sopClassUID);
489
490                // for each transfer storage setting on the datatable...
491                for (int i = 0 ; i < table.getRowCount(); i++)
492                {
493                        // get the current row/cell setting
494                        CheckboxWithHint box = (CheckboxWithHint) table.getCellData(i, 0);
495
496                        // get its UID and index (relative to the TS settings array)
497                        String uid = box.getHint();
498                        int index = sops.getTransferSettingsIndex().get(uid).intValue();
499
500                        // set the transfer storge setting value
501                        ts.setTS(box.isChecked(), index);
502                        if (accepted != null)
503                                ts.setAccepted(accepted.booleanValue());
504                }
505        }

507        /**
508         * Applies the Transfer Storage settings present on a StaticDataTable to
509         * all SOP Classes UIDs.
510         *
511         * @param accepted if all SOP Classes UIDs are accepted. Can be null and if so, the target accepted value will remain the same.
512         * @param table a StaticDataTable with the Transfer Storage settings.
513         */
514        public void setTransferSettingsForAllSOPClasses(Boolean accepted, StaticDataTable table)
515        {
516                SOPClassSettings sops = SOPClassSettings.getInstance();
517
518                HashMap<String, Boolean> allowed = new HashMap<String, Boolean>();
519
520                // for each transfer storage setting on the datatable...
521                for (int i = 0 ; i < table.getRowCount(); i++)
522                {
523                        // get the current row/cell setting
524                        CheckboxWithHint box = (CheckboxWithHint) table.getCellData(i, 0);
525
526                        // get its UID
527                        String uid = box.getHint();
528
529                        allowed.put(uid, new Boolean(box.isChecked()));
530                }
531
532                // set the transfer storage settings
533                sops.setAllSOPClassesSettings(accepted, allowed);
534        }
