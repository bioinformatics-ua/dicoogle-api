095    public DicomStorage(String[] Services, SOPList l) {
096        //just because the call to super must be the first instruction

104        //alternativeAETs.add(ServerSettingsManager.getSettings().getNodeName());
105
106        path = settings.getArchiveSettings().getMainDirectory();
107        if (path == null) {
108            path = "/dev/null";
109        }

111        this.priorityAETs = new HashSet<>(settings.getDicomServicesSettings().getPriorityAETitles());
112        LoggerFactory.getLogger(DicomStorage.class).debug("Priority C-STORE: " + this.priorityAETs);
113

116        device.setNetworkConnection(nc);
117        nae.setNetworkConnection(nc);
118
119        //we accept assoociations, this is a server
120        nae.setAssociationAcceptor(true);
121        //we support the VerificationServiceSOP
122        nae.register(new VerificationService());
123        //and the StorageServiceSOP
124        nae.register(this);

126        nae.setAETitle(settings.getDicomServicesSettings().getAETitle());

131
132        this.nae.setInstalled(true);
133        this.nae.setAssociationAcceptor(true);
134        this.nae.setAssociationInitiator(false);
135
136        int maxPDULengthReceive = settings.getDicomServicesSettings().getQueryRetrieveSettings().getMaxPDULengthReceive();
137        int maxPDULengthSend = settings.getDicomServicesSettings().getQueryRetrieveSettings().getMaxPDULengthSend();
138
139        ServerSettings s = ServerSettingsManager.getSettings();
140        this.nae.setDimseRspTimeout(60000 * 300);
141        this.nae.setIdleTimeout(60000 * 300);
142        this.nae.setMaxPDULengthReceive(maxPDULengthReceive + 1000);
143        this.nae.setMaxPDULengthSend(maxPDULengthSend + 1000);
144        this.nae.setRetrieveRspTimeout(60000 * 300);
145
146
147        // Added alternative AETitles.
148
149        naeArr = new NetworkApplicationEntity[alternativeAETs.size() + 1];
150        // Just adding the first AETitle
151        naeArr[0] = nae;
152
153        int k = 1;
154
155        for (String alternativeAET : alternativeAETs) {
156            NetworkApplicationEntity nae2 = new NetworkApplicationEntity();

158            nae2.setDimseRspTimeout(60000 * 300);
159            nae2.setIdleTimeout(60000 * 300);

167            //and the StorageServiceSOP
168            nae2.register(this);
169            nae2.setAETitle(alternativeAET);

171            Collection<String> caet = settings.getDicomServicesSettings().getAllowedAETitles();

173                String[] array = new String[caet.size()];
174                caet.toArray(array);
175                nae2.setPreferredCallingAETitle(array);
176            }
177            naeArr[k] = nae2;
178            k++;
179
180        }
181
182        // Just set the Network Application Entity array - which accepts a set of AEs.
183        device.setNetworkApplicationEntity(naeArr);
184
185
186        initTS(Services);
187    }

189    /**
190     *  Sets the tranfer capability for this execution of the storage service
191     *  @param Services Services to be supported
192     */
193    private void initTS(String[] Services) {
194        int count = list.getAccepted();
195        //System.out.println(count);
196        TransferCapability[] tc = new TransferCapability[count + 1];
197        String[] Verification = {UID.ImplicitVRLittleEndian, UID.ExplicitVRLittleEndian, UID.ExplicitVRBigEndian};
198        String[] TS;
199        TransfersStorage local;
200
201        tc[0] = new TransferCapability(UID.VerificationSOPClass, Verification, TransferCapability.SCP);
202        int j = 0;
203        for (int i = 0; i < Services.length; i++) {
204            count = 0;
205            local = list.getTS(Services[i]);
206            if (local.getAccepted()) {
207                TS = local.getVerboseTS();
208                if (TS != null) {
209
210                    tc[j + 1] = new TransferCapability(Services[i], TS, TransferCapability.SCP);
211                    j++;
212                }
213            }
214        }
215
216        // Setting the TS in all NetworkApplicationEntitys 
217        for (int i = 0; i < naeArr.length; i++) {
218
219            naeArr[i].setTransferCapability(tc);
220        }
221        nae.setTransferCapability(tc);
222    }

224    @Override
225    /**
226     * Called when a C-Store Request has been accepted
227     * Parameters defined by dcm4che2
228     */
229    public void cstore(final Association as, final int pcid, DicomObject rq, PDVInputStream dataStream, String tsuid) throws DicomServiceException, IOException {
230        //DebugManager.getSettings().debug(":: Verify Permited AETs @??C-Store Request ");
231
232        boolean permited = false;
233        Collection<String> allowedAETitles = ServerSettingsManager.getSettings()
234                .getDicomServicesSettings().getAllowedAETitles();
235        if (allowedAETitles.isEmpty()) {
236            permited = true;
237        } else {
238            permited = allowedAETitles.contains(as.getCallingAET());
239        }
240
241        if (!permited) {
242            //DebugManager.getSettings().debug("Client association NOT permited: " + as.getCallingAET() + "!");
243            System.err.println("Client association NOT permited: " + as.getCallingAET() + "!");
244            as.abort();
245
246            return;
247        } else {
248            //DebugManager.getSettings().debug("Client association permited: " + as.getCallingAET() + "!");
249            System.err.println("Client association permited: " + as.getCallingAET() + "!");
250        }
251
252        final DicomObject rsp = CommandUtils.mkRSP(rq, CommandUtils.SUCCESS);
253        onCStoreRQ(as, pcid, rq, dataStream, tsuid, rsp);
254        as.writeDimseRSP(pcid, rsp);
255        //onCStoreRSP(as, pcid, rq, dataStream, tsuid, rsp);
256    }

258    @Override
259    /**
260     * Actually do the job of saving received file on disk
261     * on this server with extras such as Lucene indexing
262     * and DICOMDIR update
263     */
264    protected void onCStoreRQ(Association as, int pcid, DicomObject rq, PDVInputStream dataStream, String tsuid, DicomObject rsp) throws IOException, DicomServiceException {
265        try {
266
267            String cuid = rq.getString(Tag.AffectedSOPClassUID);
268            String iuid = rq.getString(Tag.AffectedSOPInstanceUID);
269
270            DicomObject d = dataStream.readDataset();
271
272            d.initFileMetaInformation(cuid, iuid, tsuid);
273
274            Iterable<StorageInterface> plugins = PluginController.getInstance().getStoragePlugins(true);
275
276            URI uri = null;
277            for (StorageInterface storage : plugins) {
278                uri = storage.store(d);
279                if (uri != null) {
280                    // queue to index
281                    ImageElement element = new ImageElement();
282                    element.setCallingAET(as.getCallingAET());
283                    element.setUri(uri);
284                    queue.add(element);
285                }
286            }
287
288        } catch (IOException e) {
289            throw new DicomServiceException(rq, Status.ProcessingFailure, e.getMessage());
290        }
291    }

293    /**
294     * ImageElement is a entry of a C-STORE. For Each C-STORE RQ
295     * an ImageElement is created and are put in the queue to index.
296     *
297     * This only happens after the store in Storage Plugins.
298     *
299     * @param <E>
300     */
301    class ImageElement<E extends Comparable<? super E>>
302            implements Comparable<ImageElement<E>> {
303        private URI uri;
304        private String callingAET;
305
306        public URI getUri() {
307            return uri;
308        }
309
310        public void setUri(URI uri) {
311            this.uri = uri;
312        }
313
314        public String getCallingAET() {
315            return callingAET;
316        }
317
318        public void setCallingAET(String callingAET) {
319            this.callingAET = callingAET;
320        }
321
322        @Override
323        public int compareTo(ImageElement<E> o1) {
324            if (o1.getCallingAET().equals(this.getCallingAET()))
325                return 0;
326            else if (priorityAETs.contains(this.getCallingAET()))
327                return -1;
328            else return 1;
329        }
330    }

333    class Indexer extends Thread {
334        public Collection<IndexerInterface> plugins;
335
336        public void run() {
337            while (true) {
338                try {
339                    // Fetch an element by the queue taking into account the priorities.
340                    ImageElement element = queue.take();
341                    URI exam = element.getUri();
342                    if (exam != null) {
343                        List<Report> reports = PluginController.getInstance().indexBlocking(exam);
344                    }
345                } catch (InterruptedException ex) {
346                    LoggerFactory.getLogger(DicomStorage.class).error(ex.getMessage(), ex);
347                }
348
349            }
350
351        }
352    }

375    private String getDirectory(DicomObject d) {
376
377        String result = "UN";
378
379        String institutionName = d.getString(Tag.InstitutionName);
380        String modality = d.getString(Tag.Modality);
381        String studyDate = d.getString(Tag.StudyDate);
382        String accessionNumber = d.getString(Tag.AccessionNumber);
383        String studyInstanceUID = d.getString(Tag.StudyInstanceUID);
384        String patientName = d.getString(Tag.PatientName);
385
386        if (institutionName == null || institutionName.equals("")) {
387            institutionName = "UN_IN";
388        }
389        institutionName = institutionName.trim();
390        institutionName = institutionName.replace(" ", "");
391        institutionName = institutionName.replace(".", "");
392        institutionName = institutionName.replace("&", "");
393
394
395        if (modality == null || modality.equals("")) {
396            modality = "UN_MODALITY";
397        }
398
399        if (studyDate == null || studyDate.equals("")) {
400            studyDate = "UN_DATE";
401        } else {
402            try {
403                String year = studyDate.substring(0, 4);
404                String month = studyDate.substring(4, 6);
405                String day = studyDate.substring(6, 8);
406
407                studyDate = year + File.separator + month + File.separator + day;
408
409            } catch (Exception e) {
410                e.printStackTrace();
411                studyDate = "UN_DATE";
412            }
413        }
414
415        if (accessionNumber == null || accessionNumber.equals("")) {
416            patientName = patientName.trim();
417            patientName = patientName.replace(" ", "");
418            patientName = patientName.replace(".", "");
419            patientName = patientName.replace("&", "");
420
421            if (patientName == null || patientName.equals("")) {
422                if (studyInstanceUID == null || studyInstanceUID.equals("")) {
423                    accessionNumber = "UN_ACC";
424                } else {
425                    accessionNumber = studyInstanceUID;
426                }
427            } else {
428                accessionNumber = patientName;
429
430            }
431
432        }
433
434        result = path + File.separator + institutionName + File.separator + modality + File.separator + studyDate + File.separator + accessionNumber;
435
436        return result;
437
438    }
