021import org.apache.commons.lang3.StringUtils;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;

032import javax.xml.transform.OutputKeys;
033import javax.xml.transform.Transformer;
034import javax.xml.transform.TransformerConfigurationException;
035import javax.xml.transform.TransformerFactory;
036import javax.xml.transform.sax.SAXTransformerFactory;
037import javax.xml.transform.sax.TransformerHandler;
038import javax.xml.transform.stream.StreamResult;

048/**
049 * This class saves the list of users to XML
050 *
051 * @author Samuel Campos <samuelcampos@ua.pt>
052 */
053public class UsersXML extends DefaultHandler {
054    private static final Logger logger = LoggerFactory.getLogger(UsersXML.class);
055    Collection<User> users;
056    private boolean isUsers = false;
057
058    private String username;
059    private String Hash;
060    private boolean admin;
061    private String roles;
062
063
064    public UsersXML() {
065        username = "";
066        Hash = "";
067        roles = "";
068        admin = false;
069
070        users = new LinkedList<>();
071    }
072
073
074    @Override
075    public void startElement(String uri, String localName, String qName, Attributes attribs) {
076        if (localName.equals("Users")) {
077            isUsers = true;
078        } else if (this.isUsers && localName.equals("user")) {
079            this.username = this.resolveAttrib("username", attribs, "xp");
080            this.Hash = this.resolveAttrib("hash", attribs, "xp");
081
082            String temp = this.resolveAttrib("admin", attribs, "xp");
083            if (temp.equals("true"))
084                this.admin = true;
085            else
086                this.admin = false;
087            this.roles = this.resolveAttrib("roles", attribs, "xp");
088        }
089    }
090
091
092    @Override
093    public void endElement(String uri, String localName, String qName) {
094        if (localName.equals("Users")) {
095            isUsers = false;
096        } else if (localName.equals("user")) {
097            User u = new User(username, Hash, admin);
098            users.add(u);
099            if (roles != null) {
100                String[] rolesTmp = roles.split(",");
101                for (int i = 0; i < rolesTmp.length; i++) {
102                    Role role = RolesStruct.getInstance().getRole(rolesTmp[i]);
103                    if (role != null) u.addRole(role);
104                }
105            }
106        }
107    }
108
109    private String resolveAttrib(String attr, Attributes attribs, String defaultValue) {
110        String tmp = attribs.getValue(attr);
111        return (tmp != null) ? (tmp) : (defaultValue);
112    }
113
114
115    public Collection<User> getXML() {
116        try {
117            UserFileHandle file = new UserFileHandle();
118            byte[] xml = file.getFileContent();
119
120            if (xml == null) {
121                users = new LinkedList<>(UsersStruct.getDefaults());
122                printXML(users);
123
124                return users;
125            }
126
127            // Never throws the exception cause file not exists so need try catch
128            InputSource src = new InputSource(new ByteArrayInputStream(xml));
129            XMLReader r = XMLReaderFactory.createXMLReader();
130            r.setContentHandler(this);
131            r.parse(src);
132            return users;
133        } catch (SAXException | IOException ex) {
134            logger.error("Error reading users XML configuration file.", ex);
135        }
136        return null;
137    }
138
139    /**
140     * Print the users information to the XML file
141     */
142    public void printXML(Collection<User> users) {
143        ByteArrayOutputStream out = new ByteArrayOutputStream();
144
145        SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory.newInstance();
146        //      SAX2.0 ContentHandler.
147        TransformerHandler hd = null;
148        try {
149            hd = tf.newTransformerHandler();
150        } catch (TransformerConfigurationException ex) {
151            logger.error("Failed creating configuration object", ex);
152        }
153
154        Transformer serializer = hd.getTransformer();
155        serializer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
156        serializer.setOutputProperty(OutputKeys.METHOD, "xml");
157        serializer.setOutputProperty(OutputKeys.INDENT, "yes");
158        serializer.setOutputProperty(OutputKeys.STANDALONE, "yes");
159
160        try (PrintWriter pw = new PrintWriter(out)) {
161            StreamResult streamResult = new StreamResult(pw);
162            hd.setResult(streamResult);
163            hd.startDocument();
164
165            AttributesImpl atts = new AttributesImpl();
166            //root element
167            hd.startElement("", "", "Users", atts);
168
169            Iterator<User> us = users.iterator();
170
171            atts.clear();
172            while (us.hasNext()) {
173                User user = us.next();
174
175                atts.addAttribute("", "", "username", "", user.getUsername());
176                atts.addAttribute("", "", "hash", "", user.getPasswordHash());
177
178                String temp = "false";
179                if (user.isAdmin())
180                    temp = "true";
181                if (user.getRoles() != null && user.getRoles().size() > 0) {
182                    String roles = user.getRoles().stream()
183                            .map(Role::getName)
184                            .collect(Collectors.joining(","));
185
186                    atts.addAttribute("", "", "roles", "", roles);
187                }
188
189                atts.addAttribute("", "", "admin", "", temp);
190
191                hd.startElement("", "", "user", atts);
192                atts.clear();
193                hd.endElement("", "", "user");
194            }
195            hd.endElement("", "", "Users");
196
197            hd.endDocument();
198        } catch (SAXException ex) {
199            logger.error("Failed to parse XML file", ex);
200        } finally {
201            try {
202                UserFileHandle file = new UserFileHandle();
203                file.printFile(out.toByteArray());
204            } catch (IOException e) {
205                logger.error("Failed to write user roles file", e);
206            }
207        }
208    }
