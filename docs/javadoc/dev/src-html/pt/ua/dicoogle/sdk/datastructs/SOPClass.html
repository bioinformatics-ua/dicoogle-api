019package pt.ua.dicoogle.sdk.datastructs;
020
021import com.fasterxml.jackson.annotation.*;
022import com.fasterxml.jackson.dataformat.xml.annotation.JacksonXmlElementWrapper;
023import com.fasterxml.jackson.dataformat.xml.annotation.JacksonXmlProperty;
024
025import java.util.Collection;
026import java.util.Collections;
027import java.util.List;
028import java.util.Objects;
029
030/**
031 * @author Eduardo Pinho <eduardopinho@ua.pt>
032 * @author Luís A. Bastião Silva <bastiao@bmd-software.com>
033 */
034@JsonRootName("sop-class")
035@JsonAutoDetect(getterVisibility = JsonAutoDetect.Visibility.NONE)
036public final class SOPClass {
037
038    @JsonProperty("uid")
039    private final String uid;
040
041    @JsonProperty("ts")
042    @JacksonXmlProperty(localName = "transfer-syntaxes")
043    private final Collection<String> ts;
044
045
046
047    public SOPClass(String uid) {
048        this.uid = uid;
049        this.ts = Collections.EMPTY_LIST;
050    }
051
052    @JsonCreator
053    public SOPClass(@JsonProperty("uid")String uid, @JsonProperty("transfer-syntaxes") Collection<String> ts) {
054        this.uid = uid;
055        this.ts = ts != null ? ts : Collections.EMPTY_LIST;
056    }
057
058    public String getUID() {
059        return this.uid;
060    }
061
062    public Collection<String> getTransferSyntaxes() {
063        return this.ts;
064    }
065
066    /** Create a new, independent SOP class with the transfer syntax collection replaced.
067     */
068    public SOPClass withTS(Collection<String> ts) {
069        return new SOPClass(this.uid, ts);
070    }
071
072    /** Retrieve an SOP class with the transfer syntax collection replaced only if this SOP class
073     * has no transfer syntaxes specified, return itself otherwise.
074     */
075    public SOPClass withDefaultTS(Collection<String> ts) {
076        return this.ts.isEmpty() ? this.withTS(ts) : this;
077    }
078
079    @Override
080    public boolean equals(Object o) {
081        if (this == o) return true;
082        if (o == null || getClass() != o.getClass()) return false;
083        SOPClass sopClass = (SOPClass) o;
084        return Objects.equals(uid, sopClass.uid) &&
085                Objects.equals(ts, sopClass.ts);
086    }
087
088    @Override
089    public int hashCode() {
090        return Objects.hash(uid, ts);
091    }
092
093    @Override
094    public String toString() {
095        return "SOPClass{" +
096                "uid='" + uid + '\'' +
097                ", ts=" + ts +
098                '}';
099    }
100}
