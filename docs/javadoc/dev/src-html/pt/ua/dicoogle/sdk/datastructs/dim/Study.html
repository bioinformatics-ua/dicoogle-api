019package pt.ua.dicoogle.sdk.datastructs.dim;

021import java.net.URI;
022import java.util.ArrayList;
023import java.util.Hashtable;
024
025/**
026 * This class is a simple implementation of Study that will be returned in DIM.
027 *
028 * @author Luís A. Bastião Silva <bastiao@bmd-software.com>
029 */
030public class Study implements StudyInterface{
031
032    private Patient parent;
033    private String StudyInstanceUID ;
034    private String StudyID;
035    private String StudyData ;
036    private String StudyTime ;
037    private String AccessionNumber;
038    private String StudyDescription;
039    private String InstitutuionName;
040
041    private String operatorsName;
042    private String RequestingPhysician;
043
044    // Look in the cases where there same patient identifier with different
045    // patient names
046    private String patientName;
047
048
049    private ArrayList<Series> series = new ArrayList<Series>() ;
050    private Hashtable<String, Series> seriesHash = new Hashtable<String, Series>();
051
052    public Study(Patient patient, String StudyInstanceUID, String StudyDate)
053    {
054        this.parent = patient;
055        this.StudyInstanceUID = StudyInstanceUID ;
056        this.StudyData = StudyDate ;
057
058    }
059
060
061    public String getPatientName() {
062        return patientName;
063    }
064
065    public void setPatientName(String patientName) {
066        this.patientName = patientName;
067    }
068
069
070    /**
071     * @return the StudyInstanceUID
072     */
073    public String getStudyInstanceUID() {
074        return StudyInstanceUID;
075    }
076
077    /**
078     * @param StudyInstanceUID the StudyInstanceUID to set
079     */
080    public void setStudyInstanceUID(String StudyInstanceUID) {
081        this.StudyInstanceUID = StudyInstanceUID;
082    }
083
084    /**
085     * @return the StudyData
086     */
087    public String getStudyData() {
088        return StudyData;
089    }
090
091    /**
092     * @param StudyData the StudyData to set
093     */
094    public void setStudyData(String StudyData) {
095        this.StudyData = StudyData;
096    }
097
098
099
100    public void addSerie(Series s){
101        if (this.seriesHash.containsKey(s.getSeriesInstanceUID())){
102
103            Series existSeries = this.seriesHash.get(s.getSeriesInstanceUID());
104            ArrayList<URI> img = s.getImageList();
105            ArrayList<String> uid = s.getSOPInstanceUIDList();
106
107            int size = img.size();
108            for (int i=0;i<size;i++){
109                existSeries.addImage(img.get(i),uid.get(i));
110            }
111        }
112        else
113        {
114            this.series.add(s);
115            this.seriesHash.put(s.getSeriesInstanceUID(), s) ;
116        }
117
118    }
119
120    /**
121     * @return the series
122     */
123    public ArrayList<Series> getSeries() {
124        return series;
125    }
126
127    public Series getSeries(String seriesInstanceUID) {
128        return this.seriesHash.get(seriesInstanceUID);
129    }
130
131    /**
132     * @param series the series to set
133     */
134    public void setSeries(ArrayList<Series> series) {
135        this.series = series;
136    }
137
138    /**
139     * @return the parent
140     */
141    public Patient getParent() {
142        return parent;
143    }
144
145    /**
146     * @return the StudyTime
147     */
148    public String getStudyTime() {
149        return StudyTime;
150    }
151
152    /**
153     * @param StudyTime the StudyTime to set
154     */
155    public void setStudyTime(String StudyTime) {
156        this.StudyTime = StudyTime;
157    }
158
159    /**
160     * @return the AccessionNumber
161     */
162    public String getAccessionNumber() {
163        return AccessionNumber;
164    }
165
166    /**
167     * @param AccessionNumber the AccessionNumber to set
168     */
169    public void setAccessionNumber(String AccessionNumber) {
170        this.AccessionNumber = AccessionNumber;
171    }
172
173    /**
174     * @return the StudyDescription
175     */
176    public String getStudyDescription() {
177        return StudyDescription;
178    }
179
180    /**
181     * @param StudyDescription the StudyDescription to set
182     */
183    public void setStudyDescription(String StudyDescription) {
184        this.StudyDescription = StudyDescription;
185    }
186
187    /**
188     * @return the StudyID
189     */
190    public String getStudyID() {
191        return StudyID;
192    }
193
194    /**
195     * @param StudyID the StudyID to set
196     */
197    public void setStudyID(String StudyID) {
198        this.StudyID = StudyID;
199    }
200
201    /**
202     * @return the InstitutuionName
203     */
204    public String getInstitutuionName() {
205        return InstitutuionName;
206    }
207
208    /**
209     * @param InstitutuionName the InstitutuionName to set
210     */
211    public void setInstitutuionName(String InstitutuionName) {
212        this.InstitutuionName = InstitutuionName;
213    }
214
215
216    /**
217     * @return the operatorsName
218     */
219    public String getOperatorsName() {
220        return operatorsName;
221    }
222
223    /**
224     * @param operatorsName the operatorsName to set
225     */

227        this.operatorsName = operatorsName;
228    }

231     * @return the RequestingPhysician
232     */
233    public String getRequestingPhysician() {
234        return RequestingPhysician;
235    }
236
237    /**
238     * @param RequestingPhysician the RequestingPhysician to set
239     */
240    public void setRequestingPhysician(String RequestingPhysician) {
241        this.RequestingPhysician = RequestingPhysician;
242    }
