169    /**
170     * Adds a new Tag to the structure.
171     * 
172     * @param field
173     */
174    private synchronized void addTag(TagValue field){           
175        TagValue tmp = this.tagValueMappings.get(field.getTagNumber());
176        if(tmp == null){
177                this.tagNameMappings.put(field.getTagNumber(), field.getName());
178                this.tagValueMappings.put(field.getTagNumber(), field);                 
179        }else{
180                //merge tag information
181                tmp.updateTagInformation(field);
182                this.tagNameMappings.put(tmp.getTagNumber(), tmp.getName());
183        }
184    }
185    
186    /**
187     * Removes a given Tag from the structure
188     * @param field
189     */
190    private synchronized void removeTag(TagValue field){        
191        this.tagNameMappings.remove(field.getTagNumber());
192        this.tagValueMappings.remove(field.getTagNumber());             
193    }
194    
195    /**
196     * Adds a DIM Field. If the field already exists its information is updated.
197     * @param field
198     */
199    public synchronized void addDIMField(TagValue field){
200        //this.dimFields.put(field.getTagNumber(), field);
201        this.nDIMFields.add(field);
202        if(!isDICOMField(field) && !isPrivateField(field))
203                addPrivateField(field);
204        addTag(field);
205    }
206    
207    /**
208     * Adds a private field. Checks if the field is already present in the DICOM fields.
209     * @param field
210     * @return
211     */
212    public synchronized boolean addPrivateField(TagValue field){
213        addTag(field);
214        if(this.nDICOMFields.contains(field)){
215                return true;
216        }       
217        this.nPrivateFields.add(field);
218        return true;
219    }
220    
221    /**
222     * Removes a Private Field. It is not possible to remove the field if it belongs to the DIM or to the DICOM group.
223     * @param tagNumber
224     * @return
225     */
226    public synchronized boolean removePrivateField(int tagNumber){
227        //this.dimFields.put(field.getTagNumber(), field);
228        TagValue tag = this.tagValueMappings.get(tagNumber);
229        if(this.nDICOMFields.contains(tag)){
230                return false;
231        }
232        
233        this.nPrivateFields.remove(tag);
234        this.nDIMFields.remove(tag);
235        removeTag(tag);
236        return true;
237    }
238    
239    /**
240     * Adds a DICOM Field.
241     * @param field
242     */
243    private synchronized void addDICOMField(TagValue field){
244        //this.dimFields.put(field.getTagNumber(), field);
245        this.nDICOMFields.add(field);
246        addTag(field);
247    }

283     * @return
284     */
285    public ArrayList<String> getDIMTagNames()
286    {
287        ArrayList<String> tagNames = new ArrayList<>(this.nDIMFields.size());
288        for(TagValue tag : this.nDIMFields){
289                tagNames.add(tag.getName());
290        }
291        return tagNames;
292    }

321    /**
322     * @return the Fields which do not belong to the DIM. (DICOM+Private)
323     */
324    public Set<TagValue> getOtherFields(){
325        
326        HashSet<TagValue> tags = new HashSet<>(this.nDICOMFields.size()+this.nPrivateFields.size());
327        for(TagValue tag : this.tagValueMappings.values()){
328                if(!isDICOMField(tag))
329                        tags.add(tag);
330                
331        }
332        return tags;
333    }
