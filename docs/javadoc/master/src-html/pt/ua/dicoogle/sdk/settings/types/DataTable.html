021import java.util.HashMap;
022import org.apache.commons.lang3.StringEscapeUtils;
023import pt.ua.dicoogle.sdk.settings.Utils;

025/**
026 * Handles a data (type) that is suposed to be on a table form.
027 * Supports multiple fields per row, and allows row coppies of the first row to
028 * be made and appended to the table. The user can also rows (except the first one).
029 *
030 * @author Ant√≥nio Novo <antonio.novo@ua.pt>
031 */
032public class DataTable implements GenericSetting
033{
034        /**
035         * Column names.
036         */
037        private String[] columns;
038        /**
039         * Cell data.
040         */
041        private Object[][] data;
042
043        /**
044         * Creates a DataTable with no columns nor cells.
045         */
046        public DataTable()
047        {
048                columns = new String[0];
049                data = new Object[0][0];
050        }
051
052        /**
053         * Creates a DataTable with (unamed) columnCount columns.
054         *
055         * @param columnCount the initial number of columns.
056         */
057        public DataTable(int columnCount)
058        {
059                columns = new String[columnCount];
060                data = new String[0][columnCount];
061        }
062
063        /**
064         * Creates a DataTable with (unamed) columnCount columns and (empty) rowCount rows.
065         *
066         * @param columnCount the initial number of columns.
067         * @param rowCount the initial number of rows.
068         */
069        public DataTable(int columnCount, int rowCount)
070        {
071                columns = new String[columnCount];
072                data = new Object[rowCount][columnCount];
073        }
074
075        /**
076         * Returns the column count on this DataTable.
077         *
078         * @return the column count on this DataTable.
079         */
080        public synchronized int getColumnCount()
081        {
082                return columns.length;
083        }
084
085        /**
086         * Returns the row count on this DataTable.
087         *
088         * @return the row count on this DataTable.
089         */
090        public synchronized int getRowCount()
091        {
092                return data.length;
093        }
094
095        /**
096         * Validates a column index for later use.
097         *
098         * @param index the index (zero based) of the target column.
099         */
100        public synchronized boolean isValidColumnIndex(int index)
101        {
102                return ((index >= 0) && (index < columns.length));
103        }
104
105        /**
106         * Validates a cell index for later use.
107         *
108         * @param row the row index (zero based) of the target cell.
109         * @param column the column index (zero based) of the target cell.
110         */
111        public synchronized boolean isValidCellIndex(int row, int column)
112        {
113                return (isValidColumnIndex(column) && ((row >= 0) && (row < data.length)));
114        }
115
116        /**
117         * Adds a new (unamed) column at the end of the column list.
118         */
119        public synchronized void addColumn()
120        {
121                // create a bigger sized array
122                String[] newColumns = new String[columns.length + 1];
123                Object[][] newData = new Object[data.length][columns.length + 1];
124
125                // copy all the previous data
126                System.arraycopy(columns, 0, newColumns, 0, columns.length);
127                for (int i = 0; i < data.length; i++)
128                        System.arraycopy(data[i], 0, newData[i], 0, columns.length);
129
130                // set the new data as the current one
131                columns = newColumns;
132                data = newData;
133        }
134
135        /**
136         * Adds a new column at the end of the column list.
137         *
138         * @param name the name/title of the new column.
139         */
140        public synchronized void addColumn(String name)
141        {
142                // create a bigger sized array
143                String[] newColumns = new String[columns.length + 1];
144                Object[][] newData = new Object[data.length][columns.length + 1];
145
146                // copy all the previous data
147                System.arraycopy(columns, 0, newColumns, 0, columns.length);
148                for (int i = 0; i < data.length; i++)
149                        System.arraycopy(data[i], 0, newData[i], 0, columns.length);
150
151                // add the new data
152                newColumns[columns.length] = name;
153
154                // set the new data as the current one
155                columns = newColumns;
156                data = newData;
157        }
158
159        /**
160         * Returns the selected column name to the desired one.
161         * Returns null if the the column index is not valid.
162         *
163         * @param column the index (zero based) of the target column.
164         * @return the selected column name to the desired one.
165         */
166        public synchronized String getColumnName(int column)
167        {
168                if (! isValidColumnIndex(column))
169                        return null;
170
171                return columns[column];
172        }
173
174        /**
175         * Changes the selected column name to the desired one.
176         * Gracefully aborts if the the column index is not valid.
177         *
178         * @param column the index (zero based) of the target column.
179         * @param name the desired new name for the column.
180         */
181        public synchronized void setColumnName(int column, String name)
182        {
183                if (! isValidColumnIndex(column))
184                        return;
185
186                columns[column] = name;
187        }
188
189        /**
190         * Adds a new (empty) row at the end of the table.
191         */
192        public synchronized void addRow()
193        {
194                // create a bigger sized array
195                Object[][] newData = new Object[data.length + 1][columns.length];
196
197                // copy all the previous data
198                for (int i = 0; i < data.length; i++)
199                        System.arraycopy(data[i], 0, newData[i], 0, columns.length);
200
201                // set the new data as the current one
202                data = newData;
203        }
204
205        /**
206         * Changes the selected cell data to the desired one.
207         * Gracefully aborts if the the cell location is not valid.
208         *
209         * @param row the row index (zero based) of the target cell.
210         * @param column the column index (zero based) of the target cell.
211         * @param data the new data for the target cell.
212         */
213        public synchronized void setCellData(int row, int column, Object data)
214        {
215                if (! isValidCellIndex(row, column))
216                        return;
217
218                this.data[row][column] = data;
219        }
220
221        /**
222         * Returns the selected cell data.
223         *
224         * @param row the row index (zero based) of the wanted cell.
225         * @param column the column index (zero based) of the wanted cell.
226         * @return null if the wanted cell is invalid, an otherwise Object.
227         */
228        public synchronized Object getCellData(int row, int column)
229        {
230                if (! isValidCellIndex(row, column))
231                        return null;
232
233                return this.data[row][column];
234        }

236        // Output formats ------------------------------------------------------
237
238        @Override
239        public synchronized String toHTMLString(String htmlElementID)
240        {
241                String result = "";
242
243                result += "<div id=\"" + htmlElementID + "\" class=\"data-table\">";
244
245                // loop through all the rows and add their cell data and column name to the result
246                for (int row = 0; row < data.length; row++)
247                {
248                        result += "<div class=\"data-table-row\">";
249
250                        for (int column = 0; column < columns.length; column++)
251                        {
252                                String columnID = Utils.getHTMLElementIDFromString(htmlElementID + " " + columns[column]);
253
254                                result += "<div style=\"margin-right: 6px; display: inline-block;\">";
255                                result +=       "<span>";
256                                result +=               StringEscapeUtils.escapeHtml4(columns[column]);
257                                result +=       "</span>";
258                                result +=       "<span style=\"padding-left: 4px;\">";
259                                result +=               Utils.getHTMLInputFromType(columnID, data[row][column], true);
260                                result +=       "</span>";
261                                result += "</div>";
262                        }
263
264                        // add a button to remove each element, except the first one
265                        result +=       "<button type=\"button\" class=\"btn btn-small btn-danger removeButton\" onclick=\"removeDataTableRow(this.parentNode.parentNode, this.parentNode);\" " + ((row == 0) ? "hidden style=\"display: none !important;\"" : "") + ">Remove</button>";
266
267                        result += "</div>";
268                }
269
270                result += "</div>";
271
272                // add one button to add another element to the table
273                result += "<button type=\"button\" class=\"btn btn-small btn-success\" onclick=\"addDataTableRow(document.getElementById('" + htmlElementID + "'));\">Add</button><br />"; // FIXME replace the getelementbyid with this.previoussibling ?!? http://www.w3schools.com/dom/prop_element_previoussibling.asp
274
275                return result;
276        }

280        /**
281         * Based on a list of http request form (names & values) and a original value object,
282         * parse the first ones to match the second ones format with the new values.
283         *
284         * @param original the original DataTable.
285         * @param params the new params, and their values, to put into the DataTable.
286         * @return a new DataTable with the updated cell information.
287         */
288        @Override
289        public DataTable fromHTTPParams(HashMap<String, String[]> params, int index, String htmlElementID)
290        {
291                // find how many rows the new params list has
292                // NOTE see the FIXME bellow to understand the need for two counters
293                int paramsCountMin = Integer.MAX_VALUE;
294                int paramsCountMax = Integer.MIN_VALUE;
295                for (String[] list : params.values()) // list all the params, and find the one that has the bigger and smaller amount of values
296                {
297                        if (list.length < paramsCountMin)
298                                paramsCountMin = list.length;
299                        if (list.length > paramsCountMax)
300                                paramsCountMax = list.length;
301                }
302
303                // create a new DataTable with the same column size as the original one and the same row count as the new params
304                DataTable result = new DataTable(this.getColumnCount(), paramsCountMax);
305
306                // add all the columns titles/names
307                for (int i = 0; i < result.getColumnCount(); i++)
308                        result.setColumnName(i, this.columns[i]);
309
310                // add all the cells information available on the original DataTable
311                for (int j = 0; j < result.getColumnCount(); j++)
312                {
313                        String columnID = Utils.getHTMLElementIDFromString(htmlElementID + " " + result.columns[j]) + "[]";
314                        boolean hasNewValues = params.containsKey(columnID);
315
316                        for (int i = 0; i < result.getRowCount(); i++)
317                        {
318                                Object cellData = this.data[0][j];
319
320                                // if there is a new value for this cell then use it
321                                if (hasNewValues)
322                                {
323                                        cellData = Utils.convertStringValueIntoProperType(cellData, params, i, columnID);
324                                        // FIMXE if paramsCountMax is different of paramsCountMin them some column might not have all param values
325                                        //      this is true if boolean column are used, but browsers don't usually report back the uncheked boxes
326                                        //      so extra logic must be applied to fix this issue with boolean values
327                                        //      the problem relies on how do we know in what row is the value missing?
328                                }
329
330                                result.setCellData(i, j, cellData);
331                        }
332                }
333
334                // return the newly formed DataTable
335                return result;
336        }
